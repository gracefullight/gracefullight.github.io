ThrottleвҖӢ

л§Ө ms л§ҲлӢӨ н•ң лІҲл§Ң нҳём¶ңлҗңлӢӨ. mousemove, scroll к°ҷмқҖ мқҙлІӨнҠёлЎң нҳём¶ңлҗҳлҠ” н•ЁмҲҳлҠ” мқҙлІӨнҠё л°ңмғқмӢң л¬ҙн•ңм • нҳём¶ңлҗҳм–ҙ м„ұлҠҘ м Җн•ҳлҘј к°Җм§Җкі  мҳӨлҠ”лҚ°, мқҙлҘј л°©м§Җн•  мҲҳ мһҲлӢӨ.

лӘЁл°”мқјм—җм„ң мҠӨнҒ¬лЎӨл§Ғ лҚ”ліҙкё°м—җ мӮ¬мҡ©н•  мҲҳ мһҲлӢӨ.

мҶҢмҠӨвҖӢ

const throttleFunction = (function () {
  "use strict";

  const timeWindow = 500; // м—¬кё°м—җ мӢңк°„(ms)мқ„ м§Җм •н•ңлӢӨ
  let lastExecution = new Date(new Date().getTime() - timeWindow);
  // ES6 мқҙн•ҳмқј кІҪмҡ° ...argsм—җ нҳём¶ңн•  parameter л§ҢнҒј мӨҖлӢӨ function(arg1, arg2...)
  const throttleFunction = function (...args) {
    // м—¬кё°м—җ лЎңм§Ғмқ„ кө¬нҳ„н•ңлӢӨ
  };

  return function () {
    if (lastExecution.getTime() + timeWindow <= new Date().getTime()) {
      lastExecution = new Date();
      return throttleFunction.apply(this, arguments);
    }
  };
})();

// мӮ¬мҡ©лІ•
throttleFunction(param1, param2);

DebounceвҖӢ

л§Ҳм§Җл§ү нҳём¶ңлЎңл¶Җн„° ms нӣ„м—җ н•ЁмҲҳлҘј н•ңлІҲ нҳём¶ңн•ңлӢӨ. м§Җм—°лҗң нҳём¶ңмқ„ н•  мҲҳ мһҲкІҢ н•ҙмЈјлҠ”лҚ°, нҳём¶ңмқҙ л°ҳліөлҗҳлҠ” лҸҷм•ҲмқҖ мӢӨн–үмқ„ л°©м§Җн•ҳкі , нҳём¶ңмқҙ л©Ҳм¶ҳ л’Ө м§Җм •н•ң ms нӣ„м—җ н•ЁмҲҳлҘј мӢӨн–үн•ҙ м„ұлҠҘ м Җн•ҳлҘј л§үмқ„ мҲҳ мһҲлӢӨ.

кІҖмғү мһҗлҸҷмҷ„м„ұ кё°лҠҘм—җ м Ғн•©н•ҳлӢӨ.

мҶҢмҠӨвҖӢ

const debounceFunction = (function () {
  "use strict";

  const timeWindow = 500; // м—¬кё°м—җ мӢңк°„(ms)мқ„ м§Җм •н•ңлӢӨ
  let timeout;
  // ES6 мқҙн•ҳмқј кІҪмҡ° ...argsм—җ нҳём¶ңн•  parameter л§ҢнҒј мӨҖлӢӨ function(arg1, arg2...)
  const debounceFunction = function (...args) {
    // м—¬кё°м—җ лЎңм§Ғмқ„ кө¬нҳ„н•ңлӢӨ
  };

  return function () {
    // eslint-disable-next-line @typescript-eslint/no-this-alias
    const context = this;
    const args = arguments;

    clearTimeout(timeout);
    timeout = setTimeout(function () {
      debounceFunction.apply(context, args);
    }, timeWindow);
  };
})();

// мӮ¬мҡ©лІ•
debounceFunction(param1, param2);

мҳҲм ң - jQeury мў…мҶҚнҳ•вҖӢ

(function (global, factory) {
  if (typeof define === "function" && define.amd) {
    // AMD нҷҳкІҪ
    define(["exports", "jquery"], factory);
  } else if (
    typeof exports === "object" &&
    typeof exports.nodeName !== "string"
  ) {
    // CommonJS нҷҳкІҪ
    factory(exports, require("jquery"));
  } else {
    // мқјл°ҳ лёҢлқјмҡ°м Җ
    factory((global.module1 = global.module1 || {}), global.$);
  }

  // module1мқҳ private кё°лҠҘ кө¬нҳ„

  // exportsм—җ public functionмқ„ л¶ҷнҳҖмЈјл©ҙ лҗңлӢӨ.

});

function jsonp(url, callback) {
  const callbackName = "jsonp_" + Math.round(100000 * Math.random());
  window[callbackName] = function (data) {
    delete window[callbackName];
    document.body.removeChild(script);
    callback(data);
  };

  const script = document.createElement("script");
  script.src =
    url + (url.indexOf("?") >= 0 ? "&" : "?") + "callback=" + callbackName;
  document.body.appendChild(script);
}

function deepFreeze(obj) {
  if (!Object.isFrozen(obj)) {
    Object.freeze(obj);
  }

  for (const key in obj) {
    if (
      {}.hasOwnProperty.call(obj, key) ||
      !(
        typeof obj[key] === "function" ||
        (typeof obj[key] === "object" && !!obj[key])
      )
    ) {
      continue;
    }

    deepFreeze(obj[key]);
  }
}
