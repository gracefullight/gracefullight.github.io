 * мқҙ мһ‘м—…мқҖ м• н”ҢлҰ¬мјҖмқҙм…ҳмқҳ лӘЁл“  лӘЁл“Ҳкіј к°Ғ лӘЁл“Ҳмқҙ м„ңлЎң м–ҙл–»кІҢ мқҳмЎҙн•ҳлҠ”м§Җ нҢҢм•…н•  л•Ңк№Ңм§Җ л°ҳліөлҗ©лӢҲлӢӨ.
 *
 * мқҙлҹ¬н•ң н”„лЎңм қнҠём—җ лҢҖн•ң мқҙн•ҙлҘј мў…мҶҚм„ұ к·ёлһҳн”„лқј л¶ҖлҰ…лӢҲлӢӨ.
 *
 * мқҙ мҳҲм ңм—җм„ңлҠ” мў…мҶҚм„ұ к·ёлһҳн”„лҘј л§Ңл“Өкі  мқҙ к·ёлһҳн”„лҘј мӮ¬мҡ©н•ҳм—¬ лӘЁл“  лӘЁл“Ҳл“Өмқ„ н•ҳлӮҳмқҳ лІҲл“ӨлЎң нҢЁнӮӨм§• н•©лӢҲлӢӨ.
 * к·ёлҹј мӢңмһ‘н•ҙ ліҙкІ мҠөлӢҲлӢӨ :)
 *
 * м°ёкі : мқҙ мҳҲм ңлҠ” л§Өмҡ° лӢЁмҲңнҷ”лҗҳм–ҙ мһҲмҠөлӢҲлӢӨ. мҲңнҷҳ м°ёмЎ°, мәҗмӢұ лӘЁл“Ҳ, нҢҢмӢұ мөңм Ғнҷ” л“ұм—җ лҢҖн•ң лӮҙмҡ©мқҖ мғқлһө
 * н•ҳм—¬ к°ҖлҠҘн•ңк°Җ лӢЁмҲңн•ҳкІҢ л§Ңл“Өм—ҲмҠөлӢҲлӢӨ.
 */

const fs = require("fs");
const path = require("path");
const babylon = require("babylon");
const traverse = require("babel-traverse").default;
const { transformFromAst } = require("babel-core");

let ID = 0;

// мҡ°м„  file pathлҘј л°ӣлҠ” н•ЁмҲҳлҘј мғқм„ұн•ҳкі 
// нҢҢмқјмқ„ лӮҙмҡ©мқ„ мқҪкі , мў…мҶҚм„ұмқ„ м¶”м¶ңн•©лӢҲлӢӨ.
function createAsset(filename) {
  // нҢҢмқјмқҳ лӮҙмҡ©мқ„ л¬ёмһҗм—ҙлЎң мқҪмҠөлӢҲлӢӨ.
  const content = fs.readFileSync(filename, "utf-8");

  // мқҙм ң мқҙ нҢҢмқјмқҙ м–ҙл–Ө нҢҢмқјм—җ мў…мҶҚлҗҳлҠ”м§Җ м•Ңм•„ліҙкІ мҠөлӢҲлӢӨ. мҡ°лҰ¬лҠ” import л¬ёмһҗм—ҙмқ„ ліҙкі  мқҳмЎҙм„ұмқ„
  // нҢҢм•…н•  мҲҳ мһҲмҠөлӢҲлӢӨ н•ҳм§Җл§Ң, мқҙкІғмқҖ лӢЁмҲңн•ң м ‘к·јлІ•мқҙм–ҙм„ң, лҢҖмӢ м—җ мһҗл°”мҠӨнҒ¬лҰҪнҠё нҢҢм„ңлҘј мӮ¬мҡ©н•ҳкІ мҠөлӢҲлӢӨ.

  // мһҗл°”мҠӨнҒ¬лҰҪнҠё нҢҢм„ңл“ӨмқҖ мһҗл°”мҠӨнҒ¬лҰҪнҠё мҪ”л“ңлҘј мқҪкі  мқҙн•ҙн•  мҲҳ мһҲлҸ„лЎқ лҸ„мҷҖмЈјлҠ” нҲҙмһ…лӢҲлӢӨ.
  // нҢҢм„ңлҠ” AST(abstract syntax tree)лқјлҠ” мўҖлҚ” м¶”мғҒнҷ”лҗң лӘЁлҚёмқ„ мғқм„ұн•©лӢҲлӢӨ.
  //
  // ASTм—җ лҢҖн•ҙ мқҙн•ҙн•ҳл Өл©ҙ AST Explorer(https://astexplorer.net)мқ„ кјӯ ліҙкё°лҘј к°•л Ҙн•ҳкІҢ м¶”мІңн•©лӢҲлӢӨ.
  // ASTк°Җ м–ҙл–»кІҢ мқҙлЈЁм–ҙм ё мһҲлҠ”м§Җ нҷ•мқён•  мҲҳ мһҲмҠөлӢҲлӢӨ.
  //
  // ASTлҠ” мҡ°лҰ¬мқҳ мҪ”л“ңм—җ лҢҖн•ҙ л§ҺмқҖ м •ліҙлҘј к°Җм§Җкі  мһҲмҠөлӢҲлӢӨ. мҡ°лҰ¬лҠ” мҝјлҰ¬лҘј мқҙмҡ©н•ҳм—¬
  // мҡ°лҰ¬мқҳ мҪ”л“ңк°Җ н•ҳл ӨлҠ” мқјм—җ лҢҖн•ҙ мқҙн•ҙн•  мҲҳ мһҲмҠөлӢҲлӢӨ.
  const ast = babylon.parse(content, {
    sourceType: "module",
  });

  // мқҙ л°°м—ҙмқҖ нҳ„мһ¬ лӘЁл“Ҳмқҳ мқҳмЎҙм„ұмқ„ мғҒлҢҖ кІҪлЎңлЎң к°Җм§Җкі  мһҲмқ„ кІғмһ…лӢҲлӢӨ.
  const dependencies = [];

  // мҡ°лҰ¬лҠ” AST мҲңнҡҢлҘј нҶөн•ҙ к°Ғк°Ғмқҳ лӘЁл“Ҳл“Өмқҙ м–ҙл–Ө мқҳмЎҙм„ұмқ„ к°Җм§Җкі  мһҲлҠ”м§Җ мқҙн•ҙн•ҳл Ө н•©лӢҲлӢӨ.
  // мқҙкІғмқ„ нҶөн•ҙ ASTм•Ҳм—җм„ң лӘЁл“  import keyword м„ м–ёмқ„ нҢҢм•…н•  мҲҳ мһҲмҠөлӢҲлӢӨ.
  traverse(ast, {
    // ECMAScript лӘЁл“Ҳл“ӨмқҖ м •м ҒмқҙлҜҖлЎң л§Өмҡ° нҢҢм•…н•ҳкё° мүҪмҠөлӢҲлӢӨ.мқҙлҠ” ліҖмҲҳлҘј к°Җм ёмҳ¬ мҲҳ м—Ҷкұ°лӮҳ мЎ°кұҙл¶ҖлЎң
    // лӢӨлҘё лӘЁл“Ҳмқ„ к°Җм ёмҳ¬ мҲҳ м—ҶмқҢмқ„ мқҳлҜён•©лӢҲлӢӨ. import кө¬л¶„мқ„ ліј л•Ң л§ҲлӢӨ м№ҙмҡҙнҢ…мқ„ н•ҳкі  мқҳмЎҙм„ұмқ„ к°Җм§Җкі 
    // мһҲлҠ” кІғмңјлЎң к°„мЈј н•  мҲҳ мһҲмҠөлӢҲлӢӨ.
    ImportDeclaration: ({ node }) => {
      // import кө¬л¬ёл§ҲлӢӨ dependencies л°°м—ҙм—җ к°’мқ„ м¶”к°Җн•©лӢҲлӢӨ.
      dependencies.push(node.source.value);
    },
  });

  // лҳҗн•ң к°„лӢЁн•ң м№ҙмҡҙн„°лҘј мқҙмҡ©н•ҳм—¬ мқҙ лӘЁл“Ҳм—җ кі мң  мӢқлі„мһҗлҘј н• лӢ№н•©лӢҲлӢӨ.
  const id = ID++;

  // мҡ°лҰ¬лҠ” мқјл¶Җ лёҢлқјмҡ°м Җм—җм„ңл§Ң м§Җмӣҗн•ҳлҠ” ECMAScript moduleл“ӨмқҙлӮҳ кё°лҠҘл“Өмқ„ мӮ¬мҡ©н•  к°ҖлҠҘм„ұлҸ„ мһҲмҠөлӢҲлӢӨ.
  // мҡ°лҰ¬к°Җ л§Ңл“ңлҠ” лІҲл“Өмқҙ лӘЁл“  лёҢлқјмҡ°м Җм—җм„ң лҸҢм•„к°ҖлҸ„лЎқ Babelмқ„ мқҙмҡ©н•ҙм„ң transpileн•  мҲҳ мһҲмҠөлӢҲлӢӨ
  // (https://babeljs.io м°ёкі ).
  //
  // `presets` мҳөм…ҳмқҖ Babelмқҙ м–ҙл–»кІҢ мҡ°лҰ¬ мҪ”л“ңлҘј л°”кҝҖм§Җм—җ лҢҖн•ҙ кІ°м •н•©лӢҲлӢӨ. мҡ°лҰ¬лҠ” `babel-preset-env`
  // лҘј мқҙмҡ©н•ҳм—¬ лҢҖл¶Җл¶„мқҳ лёҢлқјмҡ°м Җм—җм„ң мҡ°лҰ¬мқҳ мҪ”л“ңлҘј мӮ¬мҡ©н•  мҲҳ мһҲлҸ„лЎқ л°”кҫёлҸ„лЎқ н•ҳкІ мҠөлӢҲлӢӨ.
  const { code } = transformFromAst(ast, null, {
    presets: ["env"],
  });

  // мқҙ лӘЁл“Ҳм—җ лҢҖн•ң м •ліҙлҘј return н•©лӢҲлӢӨ.
  return {
    id,
    filename,
    dependencies,
    code,
  };
}

// мқҙм ң лӢЁмқј лӘЁл“Ҳмқҳ мў…мҶҚм„ұмқ„ м¶”м¶ңн•  мҲҳ мһҲмңјлҜҖлЎң, entry fileмқҳ мқҳмЎҙм„ұмқ„ м¶”м¶ңн•ҳлҠ” кІғл¶Җн„° мӢңмһ‘н•ҳкІ мҠөлӢҲлӢӨ.
// мқҙ мһ‘м—…мқҖ м• н”ҢлҰ¬мјҖмқҙм…ҳмқҳ лӘЁл“  лӘЁл“Ҳкіј к°Ғ лӘЁл“Ҳмқҙ м„ңлЎң м–ҙл–»кІҢ мқҳмЎҙн•ҳлҠ”м§ҖлҘј нҢҢм•…н•  л•Ңк№Ңм§Җ кі„мҶҚ м§„н–үн•  кІғмһ…лӢҲлӢӨ.
// мқҙ мһ‘м—…мқҳ кІ°кіјл¬јмқ„ мқҳмЎҙм„ұ к·ёлһҳн”„лқј л¶ҖлҰ…лӢҲлӢӨ.
function createGraph(entry) {
  // entry fileл¶Җн„° л¶„м„қмқ„ мӢңмһ‘н•©лӢҲлӢӨ.
  const mainAsset = createAsset(entry);

  // queueлҘј мӮ¬мҡ©н•ҙм„ң лӘЁл“  assetмқҳ мқҳмЎҙм„ұмқ„ л¶„м„қн•ҳлҸ„лЎқ н•ҳкІ мҠөлӢҲлӢӨ. мқҙ мһ‘м—…мқ„ мң„н•ҙ
  // entry assetмқ„ к°Җм§Җкі  мһҲлҠ” л°°м—ҙмқ„ м •мқҳн•©лӢҲлӢӨ.
  const queue = [mainAsset];

  // м—¬кё°м„ң queueмқҳ л°ҳліөмқ„ мң„н•ҙ `for ... of` л°ҳліөл¬ёмқ„ мӮ¬мҡ©н•©лӢҲлӢӨ. мІҳмқҢм—җлҠ” queueк°Җ assetмқ„ н•ҳлӮҳл§Ң
  // к°Җм§Җкі  мһҲм§Җл§Ң мһ‘м—…мқҙ л°ҳліөлҗҳлҠ” лҸҷм•Ҳм—җ мғҲлЎңмҡҙ assetл“Өмқ„ queueм—җ м¶”к°Җн•©лӢҲлӢӨ. мқҙ л°ҳліөл¬ёмқҖ queueк°Җ
  // л№„м–ҙм§Ҳ л•Ң к№Ңм§Җ кі„мҶҚлҗ©лӢҲлӢӨ.
  for (const asset of queue) {
    // лӘЁл“  assetл“ӨмқҖ мқҳмЎҙм„ұмқҙ мһҲлҠ” лӘЁл“Ҳм—җ лҢҖн•ң мғҒлҢҖкІҪлЎңл“Өмқ„ лҰ¬мҠӨнҠёлЎң к°Җм§Җкі  мһҲмҠөлӢҲлӢӨ. мҡ°лҰ¬лҠ” к·ё лҰ¬мҠӨнҠёлҘј
    // мҲңнҡҢн•ҳл©ҙм„ң `createAsset()`н•ЁмҲҳлЎң л¶„м„қн•ҳкі , м•„лһҳ к°қмІҙлҘј нҶөн•ҳм—¬ лӘЁл“Ҳл“Өмқҳ мқҳмЎҙм„ұмқ„ м¶”мІҷн•  кІғмһ…лӢҲлӢӨ.
    asset.mapping = {};

    // мқҙкІғмқҖ мқҙ лӘЁл“Ҳмқҙ мһҲлҠ” л””л үнҶ лҰ¬мһ…лӢҲлӢӨ.
    const dirname = path.dirname(asset.filename);

    // мў…мҶҚм„ұм—җ лҢҖн•ң мғҒлҢҖ кІҪлЎң лҰ¬мҠӨнҠёлҘј мҲңнҡҢн•©лӢҲлӢӨ.
    asset.dependencies.forEach((relativePath) => {
      // `createAsset()` н•ЁмҲҳлҠ” м ҲлҢҖ кІҪлЎңк°Җ н•„мҡ”н•©лӢҲлӢӨ. dependencies л°°м—ҙмқҖ мғҒлҢҖ кІҪлЎңлҘј к°Җм§Җкі 
      // мһҲлҠ” л°°м—ҙмһ…лӢҲлӢӨ. мқҙлҹ¬н•ң кІҪлЎңл“ӨмқҖ лӘЁл“Ҳмқҙ importлҗң fileм—җ л”°лқј мғҒлҢҖм Ғмһ…лӢҲлӢӨ. л”°лқјм„ң л¶ҖлӘЁ assetмқҳ
      // кІҪлЎңлҘј мқҙмҡ©н•ҙм„ң мғҒлҢҖ кІҪлЎңлҘј м ҲлҢҖкІҪлЎңлЎң л°”кҝ”м•ј н•©лӢҲлӢӨ.
      const absolutePath = path.join(dirname, relativePath);

      // assetмқҳ лӮҙмҡ©мҡё л¶„м„қн•ҳкі , лӮҙмҡ©мқ„ мқҪкі , мқҳмЎҙм„ұмқ„ м¶”м¶ңн•©лӢҲлӢӨ.
      const child = createAsset(absolutePath);

      // `asset`мқҳ мқҳмЎҙм„ұмқҖ `child`м—җкІҢ лӢ¬л ӨмһҲмҠөлӢҲлӢӨ. мҡ°лҰ¬лҠ” `mapping` к°қмІҙм—җ relativePathмҷҖ child.idлҘј
      // мқҙмҡ©н•ҙм„ң кҙҖкі„лҘј н‘ңнҳ„н•  мҲҳ мһҲмҠөлӢҲлӢӨ.
      asset.mapping[relativePath] = child.id;

      // л§Ҳм§Җл§үмңјлЎң child assetмқ„ queueм—җ м¶”к°Җн•ҳм—¬ кө¬л¬ё л¶„м„қмқҙ л°ҳліөлҗҳлҸ„лЎқ н•©лӢҲлӢӨ.
      queue.push(child);
    });
  }

  // мқҙ мӢңм җм—җм„ң queueлҠ” м• н”ҢлҰ¬мјҖмқҙм…ҳмқҳ лӘЁл“  лӘЁл“Ҳмқҙ нҸ¬н•Ёлҗң л°°м—ҙмһ…лӢҲлӢӨ. мқҙкІғмқҙ мҡ°лҰ¬к°Җ к·ёлһҳн”„лҘј н‘ңнҳ„н•ҳлҠ” л°©лІ•мһ…лӢҲлӢӨ.
  return queue;
}

// лӢӨмқҢмңјлЎң, к·ёлһҳн”„лҘј мқҙмҡ©н•ҳм—¬ лёҢлқјмҡ°м Җм—җм„ң мӢӨн–үн•  мҲҳ мһҲлҠ” лІҲл“Өмқ„ л°ҳнҷҳн•ҳлҠ” н•ЁмҲҳлҘј м •мқҳн•©лӢҲлӢӨ.
//
// мҡ°лҰ¬мқҳ лІҲл“ӨмқҖ self-invoking(мһҗмӢ мқ„ л¶ҖлҘјмҲҳ мһҲлҠ”)н•ЁмҲҳлҘј к°Җм§Җкі  мһҲмҠөлӢҲлӢӨ.
//
// (function() {})()
//
// мқҙ н•ЁмҲҳлҠ” н•ҳлӮҳмқҳ мқёмһҗл§Ң л°ӣмқ„ мҲҳ мһҲмҠөлӢҲлӢӨ: лӘЁл“  лӘЁл“Ҳмқҳ м •ліҙлҘј к°Җм§Җкі  мһҲлҠ” к·ёлһҳн”„.
function bundle(graph) {
  let modules = "";

  // мқҙ н•ЁмҲҳлҘј кө¬м„ұн•ҳкё° м „м—җ л§Өк°ң ліҖмҲҳлЎң м „лӢ¬н•  к°қмІҙлҘј л§Ңл“ӨкІ мҠөлӢҲлӢӨ. л°ҳл“ңмӢң м•Ңм•„л‘¬м•јн•  кІғмқҖ мҡ°лҰ¬к°Җ л§Ңл“ңлҠ”
  // мҠӨнҠёл§ҒмқҖ 2к°ңмқҳ мӨ‘кҙ„нҳё({})лЎң к°җмӢём ё мһҲм–ҙм•ј н•ңлӢӨлҠ” кІғмһ…лӢҲлӢӨ. мҡ°лҰ¬лҠ” лӢӨмқҢкіј к°ҷмқҖ нҸ¬л©§мңјлЎң м¶”к°Җн• 
  // кІғмһ…лӢҲлӢӨ: `key: value,`.
  graph.forEach((mod) => {
    // к·ёлһҳн”„м•Ҳм—җ мһҲлҠ” лӘЁл“  лӘЁл“Ҳл“ӨмқҖ entryлҘј к°қмІҙлЎң к°Җм§Җкі  мһҲмҠөлӢҲлӢӨ. мҡ°лҰ¬лҠ” moduleмқҳ idлҘј
    // к°’м—җ лҢҖн•ң нӮӨлЎң мӮ¬мҡ©н•©лӢҲлӢӨ.(к°Ғ лӘЁл“Ҳл§ҲлӢӨ 2к°ңмқҳ к°’мқҙ мһҲмҠөлӢҲлӢӨ.)
    //
    // м°»лІҲм§ё к°’мқҖ н•ЁмҲҳлЎң к°җмӢј к°Ғ лӘЁл“Ҳмқҳ мҪ”л“ңмһ…лӢҲлӢӨ. к·ё мқҙмң лҠ” лӘЁл“Ҳмқҳ scopeлҘј м§Җм •н•ҙм•ј н•ҳкё° л•Ңл¬ёмһ…лӢҲлӢӨ.
