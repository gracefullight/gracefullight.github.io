  // ΊΝΝΉΕεΈΛΦ AST(abstract syntax tree)Έζ╝ΈΛΦ ΉλΑΈΞΦ Ή╢ΦΉΔΒΊβΦΈΡε ΈςρΈΞ╕ΉζΕ ΉΔζΉΕ▒ΊΧσΈΜΙΈΜν.
  //
  // ASTΉΩΡ ΈΝΑΊΧ┤ Ήζ┤ΊΧ┤ΊΧαΈινΈσ┤ AST Explorer(https://astexplorer.net)ΉζΕ Ά╝φ Έ│┤Ά╕░Έξ╝ Ά░ΧΈιξΊΧαΆ▓Ν Ή╢ΦΉ▓εΊΧσΈΜΙΈΜν.
  // ASTΆ░Α ΉΨ┤ΈΨ╗Ά▓Ν Ήζ┤ΈμρΉΨ┤Ήι╕ ΉηΙΈΛΦΉπΑ ΊβΧΉζ╕ΊΧι ΉΙα ΉηΙΉΛ╡ΈΜΙΈΜν.
  //
  // ASTΈΛΦ Ήγ░ΈουΉζα Ή╜ΦΈΥεΉΩΡ ΈΝΑΊΧ┤ ΈπΟΉζΑ ΉιΧΈ│┤Έξ╝ Ά░ΑΉπΑΆ│ι ΉηΙΉΛ╡ΈΜΙΈΜν. Ήγ░ΈουΈΛΦ Ή┐╝ΈουΈξ╝ Ήζ┤ΉγσΊΧαΉΩυ
  // Ήγ░ΈουΉζα Ή╜ΦΈΥεΆ░Α ΊΧαΈινΈΛΦ Ήζ╝ΉΩΡ ΈΝΑΊΧ┤ Ήζ┤ΊΧ┤ΊΧι ΉΙα ΉηΙΉΛ╡ΈΜΙΈΜν.
  const ast = babylon.parse(content, {
    sourceType: "module",
  });

  // Ήζ┤ Έ░░ΉΩ┤ΉζΑ ΊαΕΉηυ ΈςρΈΥΙΉζα ΉζαΉκ┤ΉΕ▒ΉζΕ ΉΔΒΈΝΑ Ά▓╜ΈκεΈκε Ά░ΑΉπΑΆ│ι ΉηΙΉζΕ Ά▓ΔΉηΖΈΜΙΈΜν.
  const dependencies = [];

  // Ήγ░ΈουΈΛΦ AST ΉΙεΊγΝΈξ╝ ΊΗ╡ΊΧ┤ Ά░ΒΆ░ΒΉζα ΈςρΈΥΙΈΥνΉζ┤ ΉΨ┤ΈΨν ΉζαΉκ┤ΉΕ▒ΉζΕ Ά░ΑΉπΑΆ│ι ΉηΙΈΛΦΉπΑ Ήζ┤ΊΧ┤ΊΧαΈιν ΊΧσΈΜΙΈΜν.
  // Ήζ┤Ά▓ΔΉζΕ ΊΗ╡ΊΧ┤ ASTΉΧΙΉΩΡΉΕε ΈςρΈΥι import keyword ΉΕιΉΨ╕ΉζΕ ΊΝΝΉΧΖΊΧι ΉΙα ΉηΙΉΛ╡ΈΜΙΈΜν.
  traverse(ast, {
    // ECMAScript ΈςρΈΥΙΈΥνΉζΑ ΉιΧΉιΒΉζ┤ΈψΑΈκε ΈπνΉγ░ ΊΝΝΉΧΖΊΧαΆ╕░ ΉΚ╜ΉΛ╡ΈΜΙΈΜν.Ήζ┤ΈΛΦ Έ│ΑΉΙαΈξ╝ Ά░ΑΉι╕Ήαυ ΉΙα ΉΩΗΆ▒░ΈΓα Ήκ░Ά▒┤Έ╢ΑΈκε
    // ΈΜνΈξ╕ ΈςρΈΥΙΉζΕ Ά░ΑΉι╕Ήαυ ΉΙα ΉΩΗΉζΝΉζΕ ΉζαΈψ╕ΊΧσΈΜΙΈΜν. import Ά╡υΈ╢ΕΉζΕ Έ│╝ ΈΧΝ ΈπΙΈΜν Ή╣┤Ήγ┤ΊΝΖΉζΕ ΊΧαΆ│ι ΉζαΉκ┤ΉΕ▒ΉζΕ Ά░ΑΉπΑΆ│ι
    // ΉηΙΈΛΦ Ά▓ΔΉε╝Έκε Ά░ΕΉμ╝ ΊΧι ΉΙα ΉηΙΉΛ╡ΈΜΙΈΜν.
    ImportDeclaration: ({ node }) => {
      // import Ά╡υΈυ╕ΈπΙΈΜν dependencies Έ░░ΉΩ┤ΉΩΡ Ά░ΤΉζΕ Ή╢ΦΆ░ΑΊΧσΈΜΙΈΜν.
      dependencies.push(node.source.value);
    },
  });

  // ΈαΡΊΧε Ά░ΕΈΜρΊΧε Ή╣┤Ήγ┤ΊΕ░Έξ╝ Ήζ┤ΉγσΊΧαΉΩυ Ήζ┤ ΈςρΈΥΙΉΩΡ Ά│ιΉει ΉΜζΈ│ΕΉηΡΈξ╝ ΊΧιΈΜ╣ΊΧσΈΜΙΈΜν.
  const id = ID++;

  // Ήγ░ΈουΈΛΦ Ήζ╝Έ╢Α Έ╕ΝΈζ╝Ήγ░ΉιΑΉΩΡΉΕεΈπΝ ΉπΑΉδΡΊΧαΈΛΦ ECMAScript moduleΈΥνΉζ┤ΈΓα Ά╕░ΈΛξΈΥνΉζΕ ΉΓυΉγσΊΧι Ά░ΑΈΛξΉΕ▒ΈΠΕ ΉηΙΉΛ╡ΈΜΙΈΜν.
  // Ήγ░ΈουΆ░Α ΈπΝΈΥεΈΛΦ Έ▓ΙΈΥνΉζ┤ ΈςρΈΥι Έ╕ΝΈζ╝Ήγ░ΉιΑΉΩΡΉΕε ΈΠΝΉΧΕΆ░ΑΈΠΕΈκζ BabelΉζΕ Ήζ┤ΉγσΊΧ┤ΉΕε transpileΊΧι ΉΙα ΉηΙΉΛ╡ΈΜΙΈΜν
  // (https://babeljs.io Ή░╕Ά│ι).
  //
  // `presets` Ήα╡ΉΖαΉζΑ BabelΉζ┤ ΉΨ┤ΈΨ╗Ά▓Ν Ήγ░Έου Ή╜ΦΈΥεΈξ╝ Έ░ΦΆ┐ΑΉπΑΉΩΡ ΈΝΑΊΧ┤ Ά▓░ΉιΧΊΧσΈΜΙΈΜν. Ήγ░ΈουΈΛΦ `babel-preset-env`
  // Έξ╝ Ήζ┤ΉγσΊΧαΉΩυ ΈΝΑΈ╢ΑΈ╢ΕΉζα Έ╕ΝΈζ╝Ήγ░ΉιΑΉΩΡΉΕε Ήγ░ΈουΉζα Ή╜ΦΈΥεΈξ╝ ΉΓυΉγσΊΧι ΉΙα ΉηΙΈΠΕΈκζ Έ░ΦΆ╛╕ΈΠΕΈκζ ΊΧαΆ▓ιΉΛ╡ΈΜΙΈΜν.
  const { code } = transformFromAst(ast, null, {
    presets: ["env"],
  });

  // Ήζ┤ ΈςρΈΥΙΉΩΡ ΈΝΑΊΧε ΉιΧΈ│┤Έξ╝ return ΊΧσΈΜΙΈΜν.
  return {
    id,
    filename,
    dependencies,
    code,
  };
}

// Ήζ┤Ήιε ΈΜρΉζ╝ ΈςρΈΥΙΉζα ΉλΖΉΗΞΉΕ▒ΉζΕ Ή╢ΦΉ╢εΊΧι ΉΙα ΉηΙΉε╝ΈψΑΈκε, entry fileΉζα ΉζαΉκ┤ΉΕ▒ΉζΕ Ή╢ΦΉ╢εΊΧαΈΛΦ Ά▓ΔΈ╢ΑΊΕ░ ΉΜεΉηΣΊΧαΆ▓ιΉΛ╡ΈΜΙΈΜν.
// Ήζ┤ ΉηΣΉΩΖΉζΑ ΉΧιΊΦΝΈουΉ╝ΑΉζ┤ΉΖαΉζα ΈςρΈΥι ΈςρΈΥΙΆ│╝ Ά░Β ΈςρΈΥΙΉζ┤ ΉΕεΈκε ΉΨ┤ΈΨ╗Ά▓Ν ΉζαΉκ┤ΊΧαΈΛΦΉπΑΈξ╝ ΊΝΝΉΧΖΊΧι ΈΧΝΆ╣ΝΉπΑ Ά│ΕΉΗΞ ΉπΕΊΨΚΊΧι Ά▓ΔΉηΖΈΜΙΈΜν.
// Ήζ┤ ΉηΣΉΩΖΉζα Ά▓░Ά│╝Έυ╝ΉζΕ ΉζαΉκ┤ΉΕ▒ Ά╖╕ΈηαΊΦΕΈζ╝ Έ╢ΑΈοΖΈΜΙΈΜν.
function createGraph(entry) {
  // entry fileΈ╢ΑΊΕ░ Έ╢ΕΉΕζΉζΕ ΉΜεΉηΣΊΧσΈΜΙΈΜν.
  const mainAsset = createAsset(entry);

  // queueΈξ╝ ΉΓυΉγσΊΧ┤ΉΕε ΈςρΈΥι assetΉζα ΉζαΉκ┤ΉΕ▒ΉζΕ Έ╢ΕΉΕζΊΧαΈΠΕΈκζ ΊΧαΆ▓ιΉΛ╡ΈΜΙΈΜν. Ήζ┤ ΉηΣΉΩΖΉζΕ ΉεΕΊΧ┤
  // entry assetΉζΕ Ά░ΑΉπΑΆ│ι ΉηΙΈΛΦ Έ░░ΉΩ┤ΉζΕ ΉιΧΉζαΊΧσΈΜΙΈΜν.
  const queue = [mainAsset];

  // ΉΩυΆ╕░ΉΕε queueΉζα Έ░αΈ│╡ΉζΕ ΉεΕΊΧ┤ `for ... of` Έ░αΈ│╡Έυ╕ΉζΕ ΉΓυΉγσΊΧσΈΜΙΈΜν. Ή▓αΉζΝΉΩΡΈΛΦ queueΆ░Α assetΉζΕ ΊΧαΈΓαΈπΝ
  // Ά░ΑΉπΑΆ│ι ΉηΙΉπΑΈπΝ ΉηΣΉΩΖΉζ┤ Έ░αΈ│╡ΈΡαΈΛΦ ΈΠβΉΧΙΉΩΡ ΉΔΙΈκεΉγ┤ assetΈΥνΉζΕ queueΉΩΡ Ή╢ΦΆ░ΑΊΧσΈΜΙΈΜν. Ήζ┤ Έ░αΈ│╡Έυ╕ΉζΑ queueΆ░Α
  // Έ╣ΕΉΨ┤ΉπΙ ΈΧΝ Ά╣ΝΉπΑ Ά│ΕΉΗΞΈΡσΈΜΙΈΜν.
  for (const asset of queue) {
    // ΈςρΈΥι assetΈΥνΉζΑ ΉζαΉκ┤ΉΕ▒Ήζ┤ ΉηΙΈΛΦ ΈςρΈΥΙΉΩΡ ΈΝΑΊΧε ΉΔΒΈΝΑΆ▓╜ΈκεΈΥνΉζΕ ΈουΉΛνΊΛ╕Έκε Ά░ΑΉπΑΆ│ι ΉηΙΉΛ╡ΈΜΙΈΜν. Ήγ░ΈουΈΛΦ Ά╖╕ ΈουΉΛνΊΛ╕Έξ╝
    // ΉΙεΊγΝΊΧαΈσ┤ΉΕε `createAsset()`ΊΧρΉΙαΈκε Έ╢ΕΉΕζΊΧαΆ│ι, ΉΧΕΈηα Ά░ζΉ▓┤Έξ╝ ΊΗ╡ΊΧαΉΩυ ΈςρΈΥΙΈΥνΉζα ΉζαΉκ┤ΉΕ▒ΉζΕ Ή╢ΦΉ▓βΊΧι Ά▓ΔΉηΖΈΜΙΈΜν.
    asset.mapping = {};

    // Ήζ┤Ά▓ΔΉζΑ Ήζ┤ ΈςρΈΥΙΉζ┤ ΉηΙΈΛΦ ΈΦΦΈιΚΊΗιΈουΉηΖΈΜΙΈΜν.
    const dirname = path.dirname(asset.filename);

    // ΉλΖΉΗΞΉΕ▒ΉΩΡ ΈΝΑΊΧε ΉΔΒΈΝΑ Ά▓╜Έκε ΈουΉΛνΊΛ╕Έξ╝ ΉΙεΊγΝΊΧσΈΜΙΈΜν.
    asset.dependencies.forEach((relativePath) => {
      // `createAsset()` ΊΧρΉΙαΈΛΦ ΉιΙΈΝΑ Ά▓╜ΈκεΆ░Α ΊΧΕΉγΦΊΧσΈΜΙΈΜν. dependencies Έ░░ΉΩ┤ΉζΑ ΉΔΒΈΝΑ Ά▓╜ΈκεΈξ╝ Ά░ΑΉπΑΆ│ι
      // ΉηΙΈΛΦ Έ░░ΉΩ┤ΉηΖΈΜΙΈΜν. Ήζ┤ΈθυΊΧε Ά▓╜ΈκεΈΥνΉζΑ ΈςρΈΥΙΉζ┤ importΈΡε fileΉΩΡ ΈΦ░Έζ╝ ΉΔΒΈΝΑΉιΒΉηΖΈΜΙΈΜν. ΈΦ░Έζ╝ΉΕε Έ╢ΑΈςρ assetΉζα
      // Ά▓╜ΈκεΈξ╝ Ήζ┤ΉγσΊΧ┤ΉΕε ΉΔΒΈΝΑ Ά▓╜ΈκεΈξ╝ ΉιΙΈΝΑΆ▓╜ΈκεΈκε Έ░ΦΆ┐ΦΉΧ╝ ΊΧσΈΜΙΈΜν.
      const absolutePath = path.join(dirname, relativePath);

      // assetΉζα ΈΓ┤ΉγσΉγ╕ Έ╢ΕΉΕζΊΧαΆ│ι, ΈΓ┤ΉγσΉζΕ Ήζ╜Ά│ι, ΉζαΉκ┤ΉΕ▒ΉζΕ Ή╢ΦΉ╢εΊΧσΈΜΙΈΜν.
      const child = createAsset(absolutePath);

      // `asset`Ήζα ΉζαΉκ┤ΉΕ▒ΉζΑ `child`ΉΩΡΆ▓Ν ΈΜυΈινΉηΙΉΛ╡ΈΜΙΈΜν. Ήγ░ΈουΈΛΦ `mapping` Ά░ζΉ▓┤ΉΩΡ relativePathΉβΑ child.idΈξ╝
      // Ήζ┤ΉγσΊΧ┤ΉΕε Ά┤ΑΆ│ΕΈξ╝ ΊΣεΊαΕΊΧι ΉΙα ΉηΙΉΛ╡ΈΜΙΈΜν.
      asset.mapping[relativePath] = child.id;

      // ΈπΙΉπΑΈπΚΉε╝Έκε child assetΉζΕ queueΉΩΡ Ή╢ΦΆ░ΑΊΧαΉΩυ Ά╡υΈυ╕ Έ╢ΕΉΕζΉζ┤ Έ░αΈ│╡ΈΡαΈΠΕΈκζ ΊΧσΈΜΙΈΜν.
      queue.push(child);
    });
  }

  // Ήζ┤ ΉΜεΉιΡΉΩΡΉΕε queueΈΛΦ ΉΧιΊΦΝΈουΉ╝ΑΉζ┤ΉΖαΉζα ΈςρΈΥι ΈςρΈΥΙΉζ┤ ΊΠυΊΧρΈΡε Έ░░ΉΩ┤ΉηΖΈΜΙΈΜν. Ήζ┤Ά▓ΔΉζ┤ Ήγ░ΈουΆ░Α Ά╖╕ΈηαΊΦΕΈξ╝ ΊΣεΊαΕΊΧαΈΛΦ Έ░σΈ▓ΧΉηΖΈΜΙΈΜν.
  return queue;
}

// ΈΜνΉζΝΉε╝Έκε, Ά╖╕ΈηαΊΦΕΈξ╝ Ήζ┤ΉγσΊΧαΉΩυ Έ╕ΝΈζ╝Ήγ░ΉιΑΉΩΡΉΕε ΉΜνΊΨΚΊΧι ΉΙα ΉηΙΈΛΦ Έ▓ΙΈΥνΉζΕ Έ░αΊβαΊΧαΈΛΦ ΊΧρΉΙαΈξ╝ ΉιΧΉζαΊΧσΈΜΙΈΜν.
//
// Ήγ░ΈουΉζα Έ▓ΙΈΥνΉζΑ self-invoking(ΉηΡΉΜιΉζΕ Έ╢ΑΈξ╝ΉΙα ΉηΙΈΛΦ)ΊΧρΉΙαΈξ╝ Ά░ΑΉπΑΆ│ι ΉηΙΉΛ╡ΈΜΙΈΜν.
//
// (function() {})()
//
// Ήζ┤ ΊΧρΉΙαΈΛΦ ΊΧαΈΓαΉζα Ήζ╕ΉηΡΈπΝ Έ░δΉζΕ ΉΙα ΉηΙΉΛ╡ΈΜΙΈΜν: ΈςρΈΥι ΈςρΈΥΙΉζα ΉιΧΈ│┤Έξ╝ Ά░ΑΉπΑΆ│ι ΉηΙΈΛΦ Ά╖╕ΈηαΊΦΕ.
function bundle(graph) {
  let modules = "";

  // Ήζ┤ ΊΧρΉΙαΈξ╝ Ά╡υΉΕ▒ΊΧαΆ╕░ ΉιΕΉΩΡ ΈπνΆ░ε Έ│ΑΉΙαΈκε ΉιΕΈΜυΊΧι Ά░ζΉ▓┤Έξ╝ ΈπΝΈΥνΆ▓ιΉΛ╡ΈΜΙΈΜν. Έ░αΈΥεΉΜε ΉΧΝΉΧΕΈΣυΉΧ╝ΊΧι Ά▓ΔΉζΑ Ήγ░ΈουΆ░Α ΈπΝΈΥεΈΛΦ
  // ΉΛνΊΛ╕ΈπΒΉζΑ 2Ά░εΉζα ΉνΣΆ┤ΕΊα╕({})Έκε Ά░ΡΉΜ╕Ήι╕ ΉηΙΉΨ┤ΉΧ╝ ΊΧεΈΜνΈΛΦ Ά▓ΔΉηΖΈΜΙΈΜν. Ήγ░ΈουΈΛΦ ΈΜνΉζΝΆ│╝ Ά░βΉζΑ ΊΠυΈσπΉε╝Έκε Ή╢ΦΆ░ΑΊΧι
  // Ά▓ΔΉηΖΈΜΙΈΜν: `key: value,`.
  graph.forEach((mod) => {
    // Ά╖╕ΈηαΊΦΕΉΧΙΉΩΡ ΉηΙΈΛΦ ΈςρΈΥι ΈςρΈΥΙΈΥνΉζΑ entryΈξ╝ Ά░ζΉ▓┤Έκε Ά░ΑΉπΑΆ│ι ΉηΙΉΛ╡ΈΜΙΈΜν. Ήγ░ΈουΈΛΦ moduleΉζα idΈξ╝
    // Ά░ΤΉΩΡ ΈΝΑΊΧε ΊΓνΈκε ΉΓυΉγσΊΧσΈΜΙΈΜν.(Ά░Β ΈςρΈΥΙΈπΙΈΜν 2Ά░εΉζα Ά░ΤΉζ┤ ΉηΙΉΛ╡ΈΜΙΈΜν.)
    //
    // Ή░╗Έ▓ΙΉπ╕ Ά░ΤΉζΑ ΊΧρΉΙαΈκε Ά░ΡΉΜ╝ Ά░Β ΈςρΈΥΙΉζα Ή╜ΦΈΥεΉηΖΈΜΙΈΜν. Ά╖╕ Ήζ┤ΉειΈΛΦ ΈςρΈΥΙΉζα scopeΈξ╝ ΉπΑΉιΧΊΧ┤ΉΧ╝ ΊΧαΆ╕░ ΈΧΝΈυ╕ΉηΖΈΜΙΈΜν.
    // ΊΧε ΈςρΈΥΙΉΩΡΉΕε Έ│ΑΉΙαΈξ╝ ΉιΧΉζαΊΧαΈσ┤ ΈΜνΈξ╕ ΈςρΈΥΙΉζ┤ΈΓα Ά╕ΑΈκεΈ▓Ν scopeΉΩΡ ΉαΒΊΨξΉζΕ Ήμ╝ΉπΑ ΉΧΛΉΧΕΉΧ╝ ΊΧσΈΜΙΈΜν.
    //
    // transpiledΈΡε ΈςρΈΥΙΈΥνΉζΑ CommonJS ΈςρΈΥΙ ΉΜεΉΛνΊΖεΉζΕ ΉΓυΉγσΊΧσΈΜΙΈΜν:
    // ΊΧ┤ΈΜ╣ ΈςρΈΥΙ ΉΜεΉΛνΊΖεΉζΑ `require`, `module`, Ά╖╕ΈουΆ│ι `exports`Έξ╝ ΊΗ╡ΊΧ┤ ΈςρΈΥΙΊβΦ ΊΧσΈΜΙΈΜν.

        a = b
