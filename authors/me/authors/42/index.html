  // Вџ░вдгвіћ AST ВѕюьџївЦ╝ ьєхьЋ┤ Ж░ЂЖ░ЂВЮў вфевЊѕвЊцВЮ┤ Вќ┤вќц ВЮўВА┤Вё▒ВЮё Ж░ђВДђЖ│а ВъѕвіћВДђ ВЮ┤ьЋ┤ьЋўвац ьЋЕвІѕвІц.
  // ВЮ┤Ж▓ЃВЮё ьєхьЋ┤ ASTВЋѕВЌљВёю вфевЊа import keyword ВёаВќИВЮё ьїїВЋЁьЋа Вѕў ВъѕВіхвІѕвІц.
  traverse(ast, {
    // ECMAScript вфевЊѕвЊцВЮђ ВаЋВаЂВЮ┤в»ђвАю вДцВџ░ ьїїВЋЁьЋўЖИ░ ВЅйВіхвІѕвІц.ВЮ┤віћ в│ђВѕўвЦ╝ Ж░ђВаИВўг Вѕў ВЌєЖ▒░вѓў ВА░Ж▒┤вХђвАю
    // вІцвЦИ вфевЊѕВЮё Ж░ђВаИВўг Вѕў ВЌєВЮїВЮё ВЮўв»ИьЋЕвІѕвІц. import ЖхгвХёВЮё в│╝ вЋї вДѕвІц В╣┤Вџ┤ьїЁВЮё ьЋўЖ│а ВЮўВА┤Вё▒ВЮё Ж░ђВДђЖ│а
    // Въѕвіћ Ж▓ЃВю╝вАю Ж░ёВБ╝ ьЋа Вѕў ВъѕВіхвІѕвІц.
    ImportDeclaration: ({ node }) => {
      // import ЖхгвгИвДѕвІц dependencies в░░ВЌ┤ВЌљ Ж░њВЮё ВХћЖ░ђьЋЕвІѕвІц.
      dependencies.push(node.source.value);
    },
  });

  // вўљьЋю Ж░ёвІеьЋю В╣┤Вџ┤ьё░вЦ╝ ВЮ┤ВџЕьЋўВЌг ВЮ┤ вфевЊѕВЌљ Ж│аВюа ВІЮв│ёВъљвЦ╝ ьЋавІ╣ьЋЕвІѕвІц.
  const id = ID++;

  // Вџ░вдгвіћ ВЮ╝вХђ вИївЮ╝Вџ░ВађВЌљВёювДї ВДђВЏљьЋўвіћ ECMAScript moduleвЊцВЮ┤вѓў ЖИ░віЦвЊцВЮё ВѓгВџЕьЋа Ж░ђвіЦВё▒вЈё ВъѕВіхвІѕвІц.
  // Вџ░вдгЖ░ђ вДївЊювіћ в▓ѕвЊцВЮ┤ вфевЊа вИївЮ╝Вџ░ВађВЌљВёю вЈїВЋёЖ░ђвЈёвАЮ BabelВЮё ВЮ┤ВџЕьЋ┤Вёю transpileьЋа Вѕў ВъѕВіхвІѕвІц
  // (https://babeljs.io В░ИЖ│а).
  //
  // `presets` ВўхВЁўВЮђ BabelВЮ┤ Вќ┤вќ╗Ж▓ї Вџ░вдг ВйћвЊювЦ╝ в░ћЖ┐ђВДђВЌљ вїђьЋ┤ Ж▓░ВаЋьЋЕвІѕвІц. Вџ░вдгвіћ `babel-preset-env`
  // вЦ╝ ВЮ┤ВџЕьЋўВЌг вїђвХђвХёВЮў вИївЮ╝Вџ░ВађВЌљВёю Вџ░вдгВЮў ВйћвЊювЦ╝ ВѓгВџЕьЋа Вѕў ВъѕвЈёвАЮ в░ћЖЙИвЈёвАЮ ьЋўЖ▓аВіхвІѕвІц.
  const { code } = transformFromAst(ast, null, {
    presets: ["env"],
  });

  // ВЮ┤ вфевЊѕВЌљ вїђьЋю ВаЋв│┤вЦ╝ return ьЋЕвІѕвІц.
  return {
    id,
    filename,
    dependencies,
    code,
  };
}

// ВЮ┤Ваю вІеВЮ╝ вфевЊѕВЮў ВбЁВєЇВё▒ВЮё ВХћВХюьЋа Вѕў ВъѕВю╝в»ђвАю, entry fileВЮў ВЮўВА┤Вё▒ВЮё ВХћВХюьЋўвіћ Ж▓ЃвХђьё░ ВІюВъЉьЋўЖ▓аВіхвІѕвІц.
// ВЮ┤ ВъЉВЌЁВЮђ ВЋаьћївдгВ╝ђВЮ┤ВЁўВЮў вфевЊа вфевЊѕЖ│╝ Ж░Ђ вфевЊѕВЮ┤ ВёювАю Вќ┤вќ╗Ж▓ї ВЮўВА┤ьЋўвіћВДђвЦ╝ ьїїВЋЁьЋа вЋїЖ╣їВДђ Ж│ёВєЇ ВДёьќЅьЋа Ж▓ЃВъЁвІѕвІц.
// ВЮ┤ ВъЉВЌЁВЮў Ж▓░Ж│╝вг╝ВЮё ВЮўВА┤Вё▒ ЖиИвъўьћёвЮ╝ вХђвдЁвІѕвІц.
function createGraph(entry) {
  // entry fileвХђьё░ вХёВёЮВЮё ВІюВъЉьЋЕвІѕвІц.
  const mainAsset = createAsset(entry);

  // queueвЦ╝ ВѓгВџЕьЋ┤Вёю вфевЊа assetВЮў ВЮўВА┤Вё▒ВЮё вХёВёЮьЋўвЈёвАЮ ьЋўЖ▓аВіхвІѕвІц. ВЮ┤ ВъЉВЌЁВЮё ВюёьЋ┤
  // entry assetВЮё Ж░ђВДђЖ│а Въѕвіћ в░░ВЌ┤ВЮё ВаЋВЮўьЋЕвІѕвІц.
  const queue = [mainAsset];

  // ВЌгЖИ░Вёю queueВЮў в░ўв│хВЮё ВюёьЋ┤ `for ... of` в░ўв│хвгИВЮё ВѓгВџЕьЋЕвІѕвІц. В▓ўВЮїВЌљвіћ queueЖ░ђ assetВЮё ьЋўвѓўвДї
  // Ж░ђВДђЖ│а ВъѕВДђвДї ВъЉВЌЁВЮ┤ в░ўв│хвљўвіћ вЈЎВЋѕВЌљ ВЃѕвАюВџ┤ assetвЊцВЮё queueВЌљ ВХћЖ░ђьЋЕвІѕвІц. ВЮ┤ в░ўв│хвгИВЮђ queueЖ░ђ
  // в╣ёВќ┤ВДѕ вЋї Ж╣їВДђ Ж│ёВєЇвљЕвІѕвІц.
  for (const asset of queue) {
    // вфевЊа assetвЊцВЮђ ВЮўВА┤Вё▒ВЮ┤ Въѕвіћ вфевЊѕВЌљ вїђьЋю ВЃЂвїђЖ▓йвАювЊцВЮё вдгВіцьіИвАю Ж░ђВДђЖ│а ВъѕВіхвІѕвІц. Вџ░вдгвіћ ЖиИ вдгВіцьіИвЦ╝
    // ВѕюьџїьЋўвЕ┤Вёю `createAsset()`ьЋеВѕўвАю вХёВёЮьЋўЖ│а, ВЋёвъў Ж░ЮВ▓┤вЦ╝ ьєхьЋўВЌг вфевЊѕвЊцВЮў ВЮўВА┤Вё▒ВЮё ВХћВ▓ЎьЋа Ж▓ЃВъЁвІѕвІц.
    asset.mapping = {};

    // ВЮ┤Ж▓ЃВЮђ ВЮ┤ вфевЊѕВЮ┤ Въѕвіћ вћћваЅьєавдгВъЁвІѕвІц.
    const dirname = path.dirname(asset.filename);

    // ВбЁВєЇВё▒ВЌљ вїђьЋю ВЃЂвїђ Ж▓йвАю вдгВіцьіИвЦ╝ ВѕюьџїьЋЕвІѕвІц.
    asset.dependencies.forEach((relativePath) => {
      // `createAsset()` ьЋеВѕўвіћ Ваѕвїђ Ж▓йвАюЖ░ђ ьЋёВџћьЋЕвІѕвІц. dependencies в░░ВЌ┤ВЮђ ВЃЂвїђ Ж▓йвАювЦ╝ Ж░ђВДђЖ│а
      // Въѕвіћ в░░ВЌ┤ВъЁвІѕвІц. ВЮ┤вЪгьЋю Ж▓йвАювЊцВЮђ вфевЊѕВЮ┤ importвљю fileВЌљ вћ░вЮ╝ ВЃЂвїђВаЂВъЁвІѕвІц. вћ░вЮ╝Вёю вХђвфе assetВЮў
      // Ж▓йвАювЦ╝ ВЮ┤ВџЕьЋ┤Вёю ВЃЂвїђ Ж▓йвАювЦ╝ ВаѕвїђЖ▓йвАювАю в░ћЖ┐ћВЋ╝ ьЋЕвІѕвІц.
      const absolutePath = path.join(dirname, relativePath);

      // assetВЮў вѓ┤ВџЕВџИ вХёВёЮьЋўЖ│а, вѓ┤ВџЕВЮё ВЮйЖ│а, ВЮўВА┤Вё▒ВЮё ВХћВХюьЋЕвІѕвІц.
      const child = createAsset(absolutePath);

      // `asset`ВЮў ВЮўВА┤Вё▒ВЮђ `child`ВЌљЖ▓ї вІгвацВъѕВіхвІѕвІц. Вџ░вдгвіћ `mapping` Ж░ЮВ▓┤ВЌљ relativePathВЎђ child.idвЦ╝
      // ВЮ┤ВџЕьЋ┤Вёю Ж┤ђЖ│ёвЦ╝ ьЉюьўёьЋа Вѕў ВъѕВіхвІѕвІц.
      asset.mapping[relativePath] = child.id;

      // вДѕВДђвДЅВю╝вАю child assetВЮё queueВЌљ ВХћЖ░ђьЋўВЌг ЖхгвгИ вХёВёЮВЮ┤ в░ўв│хвљўвЈёвАЮ ьЋЕвІѕвІц.
      queue.push(child);
    });
  }

  // ВЮ┤ ВІюВаљВЌљВёю queueвіћ ВЋаьћївдгВ╝ђВЮ┤ВЁўВЮў вфевЊа вфевЊѕВЮ┤ ьЈгьЋевљю в░░ВЌ┤ВъЁвІѕвІц. ВЮ┤Ж▓ЃВЮ┤ Вџ░вдгЖ░ђ ЖиИвъўьћёвЦ╝ ьЉюьўёьЋўвіћ в░Ев▓ЋВъЁвІѕвІц.
  return queue;
}

// вІцВЮїВю╝вАю, ЖиИвъўьћёвЦ╝ ВЮ┤ВџЕьЋўВЌг вИївЮ╝Вџ░ВађВЌљВёю ВІцьќЅьЋа Вѕў Въѕвіћ в▓ѕвЊцВЮё в░ўьЎўьЋўвіћ ьЋеВѕўвЦ╝ ВаЋВЮўьЋЕвІѕвІц.
//
// Вџ░вдгВЮў в▓ѕвЊцВЮђ self-invoking(ВъљВІаВЮё вХђвЦ╝Вѕў Въѕвіћ)ьЋеВѕўвЦ╝ Ж░ђВДђЖ│а ВъѕВіхвІѕвІц.
//
// (function() {})()
//
// ВЮ┤ ьЋеВѕўвіћ ьЋўвѓўВЮў ВЮИВъљвДї в░ЏВЮё Вѕў ВъѕВіхвІѕвІц: вфевЊа вфевЊѕВЮў ВаЋв│┤вЦ╝ Ж░ђВДђЖ│а Въѕвіћ ЖиИвъўьћё.
function bundle(graph) {
  let modules = "";

  // ВЮ┤ ьЋеВѕўвЦ╝ ЖхгВё▒ьЋўЖИ░ ВаёВЌљ вДцЖ░ю в│ђВѕўвАю ВаёвІгьЋа Ж░ЮВ▓┤вЦ╝ вДївЊцЖ▓аВіхвІѕвІц. в░ўвЊюВІю ВЋїВЋёвЉгВЋ╝ьЋа Ж▓ЃВЮђ Вџ░вдгЖ░ђ вДївЊювіћ
  // ВіцьіИвДЂВЮђ 2Ж░юВЮў ВцЉЖ┤ёьўИ({})вАю Ж░љВІИВаИ ВъѕВќ┤ВЋ╝ ьЋювІцвіћ Ж▓ЃВъЁвІѕвІц. Вџ░вдгвіћ вІцВЮїЖ│╝ Ж░ЎВЮђ ьЈгвЕДВю╝вАю ВХћЖ░ђьЋа
  // Ж▓ЃВъЁвІѕвІц: `key: value,`.
  graph.forEach((mod) => {
    // ЖиИвъўьћёВЋѕВЌљ Въѕвіћ вфевЊа вфевЊѕвЊцВЮђ entryвЦ╝ Ж░ЮВ▓┤вАю Ж░ђВДђЖ│а ВъѕВіхвІѕвІц. Вџ░вдгвіћ moduleВЮў idвЦ╝
    // Ж░њВЌљ вїђьЋю ьѓцвАю ВѓгВџЕьЋЕвІѕвІц.(Ж░Ђ вфевЊѕвДѕвІц 2Ж░юВЮў Ж░њВЮ┤ ВъѕВіхвІѕвІц.)
    //
    // В░╗в▓ѕВДИ Ж░њВЮђ ьЋеВѕўвАю Ж░љВІ╝ Ж░Ђ вфевЊѕВЮў ВйћвЊюВъЁвІѕвІц. ЖиИ ВЮ┤Вюавіћ вфевЊѕВЮў scopeвЦ╝ ВДђВаЋьЋ┤ВЋ╝ ьЋўЖИ░ вЋївгИВъЁвІѕвІц.
    // ьЋю вфевЊѕВЌљВёю в│ђВѕўвЦ╝ ВаЋВЮўьЋўвЕ┤ вІцвЦИ вфевЊѕВЮ┤вѓў ЖИђвАюв▓ї scopeВЌљ ВўЂьќЦВЮё ВБ╝ВДђ ВЋіВЋёВЋ╝ ьЋЕвІѕвІц.
    //
    // transpiledвљю вфевЊѕвЊцВЮђ CommonJS вфевЊѕ ВІюВіцьЁюВЮё ВѓгВџЕьЋЕвІѕвІц:
    // ьЋ┤вІ╣ вфевЊѕ ВІюВіцьЁюВЮђ `require`, `module`, ЖиИвдгЖ│а `exports`вЦ╝ ьєхьЋ┤ вфевЊѕьЎћ ьЋЕвІѕвІц.
    // ВЮ┤ ьѓцВЏївЊювЊцВЮђ ВЮ╝в░ўВаЂВю╝вАю вИївЮ╝Вџ░ВађВЌљВёю ВѓгВџЕьЋаВѕў ВЌєВю╝в»ђвАю, Вџ░вдгВЮў ьЋеВѕўвЦ╝ ВЮ┤ВџЕьЋўВЌг ВБ╝ВъЁьЋ┤ВЋ╝ ьЋЕвІѕвІц.
    //
    // вЉљв▓ѕВДИ Ж░њВЮђ вфевЊѕЖ░ёВЮў ВЮўВА┤Вё▒ вДцьЋЉВЮё stringifyьЋўвіћ Ж▓ЃВъЁвІѕвІц. вІцВЮїЖ│╝ Ж░ЎВЮђ Ж░ЮВ▓┤ВъЁвІѕвІц.
    // { './relative/path': 1 }.
    //
    // transpiledвљю Вџ░вдгВЮў вфевЊѕвЊцВЮ┤ ВЃЂвїђЖ▓йвАюВЎђ ьЋЕЖ╗ў `require()`вЦ╝ ьўИВХюьЋўЖИ░ вЋївгИВъЁвІѕвІц. ВЮ┤ ьЋеВѕўвЦ╝ ьўИВХюьЋўвЕ┤
    // ЖиИвъўьћёВЌљВёю ВЮ┤ вфевЊѕВЮў ВЃЂвїђ Ж▓йвАюВЌљ ьЋ┤вІ╣ьЋўвіћ вфевЊѕВЮё ьЎЋВЮИьЋа Вѕў ВъѕВіхвІѕвІц.
    modules += `${mod.id}: [
      function (require, module, exports) { ${mod.code} },
      ${JSON.stringify(mod.mapping)},
    ],`;
  });

  // вДѕВДђвДЅВю╝вАю self-invoking ьЋеВѕўВЮў bodyвЦ╝ вДївЊГвІѕвІц.
  //
  // `require()` ьЋеВѕўвЦ╝ вДївЊцвЕ░ ВІюВъЉьЋўЖ▓аВіхвІѕвІц: вфевЊѕ idвЦ╝ в░ЏВЋё ВЋъВёю вДївЊа вфевЊѕ ВўцвИїВаЮьіИВЌљВёю `module`ВЮё
  // В░ЙВіхвІѕвІц. function wrapperВЎђ вДхьЋЉ Ж░ЮВ▓┤вЦ╝ Вќ╗ЖИ░ВюёьЋ┤ two-value Ж░ЮВ▓┤вЦ╝ ВЮ┤ВџЕьЋЕвІѕвІц.
  //
  // вфевЊѕВЮў ВйћвЊювіћ вфевЊѕВЮў idвЊц вїђВІа ВЃЂвїђЖ▓йвАюВЎђ ьЋеЖ╗ў `require()`ьЋеВѕўвЦ╝ ьўИВХюьЋЕвІѕвІц. Вџ░вдгЖ░ђ вДївЊа require ьЋеВѕўвіћ
  // id в░ЏВіхвІѕвІц. вўљьЋю вЉљЖ░юВЮў вфевЊѕВЮђ вЈЎВЮ╝ьЋю ВЃЂвїђ Ж▓йвАювЦ╝ ВџћЖхгьЋа Вѕў ВъѕВДђвДї, ВІцВаювАа вЉљЖ░юВЮў вІцвЦИ вфевЊѕвЊцВЮё
  // ВЮўв»ИьЋўЖ▓ї вљЕвІѕвІц.
  //
  // ВЮ┤ вгИВаювЦ╝ ьЋ┤Ж▓░ьЋўЖИ░ ВюёьЋ┤ в│ёвЈёВЮў `require` ьЋеВѕўвЦ╝ ВаюЖ│хьЋЕвІѕвІц. вфевЊѕВЮў вДхьЋЉ ВўцвИїВаЮьіИвЦ╝ ВЮ┤ВџЕьЋўВЌг ВЃЂвїђЖ▓йвАювЦ╝ idsВЌљ ьЋавІ╣ьЋЕвІѕвІц.
  // вДхьЋЉ ВўцвИїВаЮьіИвіћ ЖхгВ▓┤ВаЂВЮИ вфевЊѕВЮё Ж░ђВаИВўцЖИ░ ВюёьЋю ВџЕвЈёвАю, ВЃЂвїђ Ж▓йвАюВЎђ вфевЊѕ idsвЦ╝ вДхьЋЉьЋЕвІѕвІц.
  //
  // вДѕВДђвДЅВю╝вАю, вфевЊѕВЮ┤ require вљўВЌѕВЮё вЋї exports Ж░ЮВ▓┤ВЮў Ж░њВЮ┤ вЁИВХювљўВќ┤ВЋ╝ ьЋЕвІѕвІц. вћ░вЮ╝Вёю вфевЊѕ ВйћвЊюВЌљ ВЮўьЋ┤ в│ђьЎў вљю
  // `exports` Ж░ЮВ▓┤віћ `require()`вАю в░ўьЎўвљЕвІѕвІц.
  const result = `
    (function(modules) {
      function require(id) {
        const [fn, mapping] = modules[id];
        function localRequire(name) {
          return require(mapping[name]);
        }
        const module = { exports : {} };
        fn(localRequire, module, module.exports);
        return module.exports;
      }
      require(0);
