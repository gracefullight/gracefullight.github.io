redux, redux-thunk, redux-promise-middleware, redux-actions, redux-saga лЁёлҰ¬к°Җ лҪҖк°ңм§Ҳ м§ҖкІҪмқҙлӢӨ. кІ°көӯм—” redux-sagaлҘј мҚЁм•јл§Ң н–Ҳкі  мҷң sagaлЎң мҲҳл ҙн•ҳкІҢ лҗҳлҠ”м§Җм—җ лҢҖн•ң мӮҪм§Ҳкё°лӢӨ.

лҸ„лҢҖмІҙ лӘҮ к°ңмқҳ лҜёл“ӨмӣЁм–ҙ лқјмқҙлёҢлҹ¬лҰ¬лҘј нҢҢм•јн•ҳлҠ”м§Җ нҷ”лӮЁмқ„ нҸ¬мҠӨнҢ…н–Ҳкё°м—җ мҳӨ н”„лЎ нҠём—”л“ңлҘј кҫёлҜёлҠ”лҚ°лҠ” лҰ¬м•ЎнҠёк°Җ мөңкі м•ј лқјкі  мғқк°Ғн•ҳлҠ” л¶„л“Өм—җкІҗ л§ҲмқҢмқҙ м•Ҳл“Ө мҲҳлҸ„ мһҲлӢӨ. м •мӢ кұҙк°•мқ„ мң„н•ҙ mobx мӮ¬мҡ©мқ„ м¶”мІңл“ңлҰҪлӢҲлӢӨ.

reduxвҖӢ

н•ҳмң„, мғҒмң„ м»ҙнҸ¬л„ҢнҠём—җ лҚ°мқҙн„°лҘј propsлЎң л„ҳкІЁмЈјлҠ”кІҢ л„Ҳл¬ҙ кҙҖлҰ¬н•ҳкё° нһҳл“Өм–ҙм„ң м„ нғқн•ңлӢӨ. н•ҳлӮҳмқҳ Store(Object)м—җ SPAмқҳ лӘЁл“  лҚ°мқҙн„°лҘј ліҙкҙҖн•ңлӢӨ. лҢҖм•ҲмңјлЎңлҠ” Event Bus ComponentлҘј л§Ңл“Өл©ҙ лҗҳлҠ”лҚ°, н•ҳлӮҳмқҳ мҠӨнҶ м–ҙлҘј к°–лҠ”кІҢ к·ёкІҢ к·ёкұ°лқј кұ°мқҳ к°•м ңлҗңлӢӨ. reduxлҘј м Ғмҡ©н•ҳкё°м—җ м•һм„ң кё°мЎҙ MVC нҢЁн„ҙм—җ м©Ңл“  мӮ¬мғҒмқ„ Flex(лӢЁл°©н–Ҙ) нҢЁн„ҙмңјлЎң л°”кҫјлӢӨлҠ”кІҢ кұ°мқҳ лӮЁл¶Ғк°„ нҷ”н•© мҲҳмӨҖмқҙмҳҖлӢӨ. мқҙ л¶Җл¶„м—җ мһҲм–ҙм„  м—°мҠөл§Ңмқҙ лӢөмқҙлӢӨ. лҲҲк°җкі  мҪ”л”©н•  мҲҳ мһҲмқ„ л§ҢнҒј мҳҲм ң н”„лЎңм қнҠёлҘј л°ҳліөн•ҙліҙмһҗ. мқјлӢЁ лӘём—җ мқөмңјл©ҙ action creatorк°Җ лӯ”м§Җ мҷң dispatchлҘј н•ҙм•јн•ҳкі  reducerлЎң к°’мқ„ мІҳлҰ¬н•ҙм•ј н•ҳлҠ”м§Җк°Җ лҮҢм—җм„ңлҸ„ мқҙн•ҙк°Җ лҗҳлҠ” лҠҗлӮҢмқҙмҳҖлӢӨ. (counter мҳҲм ңлҠ” мўӢм§Җ м•ҠлӢӨкі  ліёлӢӨ. shopping cartмҳҲм ңлӮҳ todo мҳҲм ңк°Җ лҚ” мқҙн•ҙн•ҳкё° мүҪлӢӨ.)

redux-saga

м–ҙл–Ө actionмқҖ promise payloadлҘј лұүкі , м–ҙл–Ө actionмқҖ dispatchлЎң лӢӨлҘё actionмқ„ нҳём¶ңн•ҳкі  м–ҙл–Ө actionмқҖ plain objectлҘј лұүкі  мқҙлҹ° мқјкҙҖм„ұм—ҶлҠ” м§“мқ„ кі„мҶҚн•ҳлӢӨліҙл©ҙ м •л§җ л№„нҡЁмңЁм ҒмқҙлқјлҠ” лҠҗлӮҢмқ„ л°ӣлҠ”лӢӨ. м•Ҳ л°ӣмңјм…ЁлӢӨл©ҙ н•ҙліҙмӢңл©ҙ лҠҗк»ҙм§Ҳ кІғмқҙлӢӨ.

лҚ” к·№лӢЁм ҒмңјлЎң л“  мғқк°ҒмқҖ VuexлҘј мӮ¬мҡ©н•ҳл©ҙ к·ёлғҘ mapGetters, mapActions л‘җ н•ЁмҲҳл§ҢмңјлЎң м§ҒкҙҖм Ғмқҙкі  л””мһҗмқҙл„ҲлҸ„ м•Ңм•„ліј лІ•н•ң мҪ”л“ңлЎң кө¬нҳ„мқҙ к°ҖлҠҘн•ңлҚ° мҷң reactлҘј н•ҙм•јлҗҳм§Җ? лһҖ мқҳл¬ёмқҙ кі„мҶҚ л“Өм—ҲлӢӨ.

Redux-sagaлҘј м ‘н•ҳкі лҠ” л“ңл””м–ҙ state кҙҖлҰ¬мқҳ м№ңкө¬лҘј л§ҢлӮң лҠҗлӮҢмқҙмҳҖлӢӨ. к·ёл ҮлӢӨл©ҙ sagaлҠ” лӯҳк№Ң? н•ң мӨ„ н‘ңнҳ„л§Ң кё°м–өн•ҙліҙмһҗ.

sagaлһҖ actionм—җ лҢҖн•ң listenerмқҙлӢӨ. мқҢ.. м•Ўм…ҳ лҰ¬мҠӨл„Ҳкө¬лӮҳ. мқҙлІӨнҠё лҰ¬мҠӨл„Ҳк°ҷмқҖ кІғмқҙкө°.

м„Өм№ҳлҘј н•ҳкё° м „м—җ reduxмҷҖ generatorмқҳ к°ңл…җмқ„ мҷ„лІҪнһҲ мқҙн•ҙн–ҲлӢӨкі  к°Җм •н•ңлӢӨ.

const INITIAL_STATE = {
  boards: [],
};

export default (state = INITIAL_STATE, { type, payload, error }) => {
  switch (type) {
    case FETCH_BOARDS_FULFILLED:
      return {
        ...state,
        boards: payload,
      };
    case FETCH_BOARDS_REJECTED:
      return {
        ...state,
        showError: true,
        error,
      };
    default:
      return state;
  }
};

sagaвҖӢ

sagaлҠ” actionмқ„ listen(watch)н•ңлӢӨ.

saga/index.js
import { call, spawn, put, takeEvery } from "redux-saga/effects";
import * as actions from "../actions";
import axios from "axios";

function* fetchBoardsSaga() {
  // try catch finally кө¬л¬ёмңјлЎң мҳӨлҘҳ м ңм–ҙк°Җ к°ҖлҠҘн•ҳлӢӨ.
  try {
    // мқҙл¶Җл¶„мқ„ call л©”мҶҢл“ңлҘј мқҙмҡ©н•ҙ н…ҢмҠӨнҠёк°Җ мүҪкІҢ л°”кҝҖ мҲҳ мһҲлӢӨ.
    // (yeildлҘј мӮ¬мҡ©н•ҳкё° л•Ңл¬ём—җ next лӘ…л №м–ҙлЎң л°ҳліө к°ҖлҠҘн•ҳлҜҖлЎң)
    // const { data } = yield call([axios, 'get'], '/boards')
    const { data } = yield axios.get("/boards");
    yield put(actions.fetchBoardsFulfilled(data));
  } catch (error) {
    yield put(actions.fetchBoardsRejected(error.response));
  }
}

function* watchBoard() {
  // typeмқҳ actionмқҙ мӢӨн–үлҗҳл©ҙ fetchBoardsSagaлҸ„ н•ӯмғҒ(Every) мӢӨн–үн•ңлӢӨ
  yield takeEvery(FETCH_BOARDS, fetchBoardsSaga);
}

// лӘЁл“  listener(watcher)лҘј н•ҳлӮҳлЎң л¬¶м–ҙмӨҖлӢӨ.
// rootReducerм—җ л¬¶м–ҙмЈјлҠ” к·ёкІғкіј к°ҷлӢӨкі  ліҙл©ҙ лҗңлӢӨ.
export default function* root() {
  yield spawn(watchBoard);
}

мҷң watcherл“Өмқ„ spawnмңјлЎң л¬¶м–ҙм•јн•ҳлҠ”м§ҖлҠ” мқҙмҠҲм—җ лӮҳмҷҖмһҲлӢӨ. (м—¬кё°м—” мһҗлҸҷмңјлЎң sagaк°Җ мһ¬мӢңмһ‘лҗҳлҠ” нҢЁн„ҙлҸ„ мһҲлҠ”лҚ°, м•„м§Ғ нҷңмҡ©н•ҙліё м ҒмқҖ м—ҶлӢӨ)

action => saga => action => reducer лЎң м—°кІ°лҗҳлҠ” sagaк°Җ мҷ„м„ұлҗҳм—ҲлӢӨ.

store/index.js
import createSagaMiddleware, { END } from "redux-saga";
const saga = createSagaMiddleware();

export default function configureStore(initialState) {
  const store = createStore(
    rootReducer,
    initialState,
    compose(
      applyMiddleware(...middlewares),
      inDevelopment && window.devToolsExtension
        ? window.devToolsExtension()
        : (f) => f
    )
  );

  store.runSaga = saga.run;
  store.close = () => store.dispatch(END);

  return store;
}

allвҖӢ

yield кө¬л¬ёмңјлЎң кё°лӢӨлҰ¬лҠ” кұҙ мҲңм°Ём ҒмңјлЎң мӢӨн–үлҗҳкё° л•Ңл¬ём—җ, лҸҷмӢңм—җ мӢӨн–үлҗҳкі  м „л¶Җ resolveлҗҳлҠ” нҢЁн„ҙмқҙ н•„мҡ”н•ҳлӢӨл©ҙ all л©”мҶҢл“ңлҘј мӮ¬мҡ©н•ҳл©ҙ лҗңлӢӨ. (Promise.allмқ„ мғқк°Ғн•ҳл©ҙ мүҪлӢӨ)

export function* testSaga() {
  //  кё°мЎҙ нҸ¬мҠӨнҢ…л“Өм—җлҠ” мқҙл ҮкІҢ мӮ¬мҡ©н•ҳлқјлҠ” кө¬л¬ёмқҙ л§ҺмқҖлҚ°
  // deprecated кІҪкі лҘј лұүлҠ”лӢӨ
  const [response1, response2] = yield [asyncTask1(), asyncTask2()];

  // м•„лһҳмҷҖ к°ҷмқҙ мӮ¬мҡ©н•ҳмһҗ
  const [response1, response2] = yield all([asyncTask1(), asyncTask2()]);
}
