  boards: [],

  switch (type) {
    case FETCH_BOARDS_FULFILLED:
      return {
        ...state,
        boards: payload,
      };
    case FETCH_BOARDS_REJECTED:
      return {
        ...state,
        showError: true,
        error,
      };
    default:
      return state;
  }

sagaΈΛΦ actionΉζΕ listen(watch)ΊΧεΈΜν.

saga/index.js
import { call, spawn, put, takeEvery } from "redux-saga/effects";
import * as actions from "../actions";
import axios from "axios";

function* fetchBoardsSaga() {
  // try catch finally Ά╡υΈυ╕Ήε╝Έκε ΉανΈξα ΉιεΉΨ┤Ά░Α Ά░ΑΈΛξΊΧαΈΜν.
  try {
    // Ήζ┤Έ╢ΑΈ╢ΕΉζΕ call ΈσΦΉΗΝΈΥεΈξ╝ Ήζ┤ΉγσΊΧ┤ ΊΖΝΉΛνΊΛ╕Ά░Α ΉΚ╜Ά▓Ν Έ░ΦΆ┐Α ΉΙα ΉηΙΈΜν.
    // (yeildΈξ╝ ΉΓυΉγσΊΧαΆ╕░ ΈΧΝΈυ╕ΉΩΡ next ΈςΖΈι╣ΉΨ┤Έκε Έ░αΈ│╡ Ά░ΑΈΛξΊΧαΈψΑΈκε)
    // const { data } = yield call([axios, 'get'], '/boards')
    const { data } = yield axios.get("/boards");
    yield put(actions.fetchBoardsFulfilled(data));
  } catch (error) {
    yield put(actions.fetchBoardsRejected(error.response));
  }
}

function* watchBoard() {
  // typeΉζα actionΉζ┤ ΉΜνΊΨΚΈΡαΈσ┤ fetchBoardsSagaΈΠΕ ΊΧφΉΔΒ(Every) ΉΜνΊΨΚΊΧεΈΜν
  yield takeEvery(FETCH_BOARDS, fetchBoardsSaga);
}

// ΈςρΈΥι listener(watcher)Έξ╝ ΊΧαΈΓαΈκε Έυ╢ΉΨ┤ΉνΑΈΜν.
// rootReducerΉΩΡ Έυ╢ΉΨ┤Ήμ╝ΈΛΦ Ά╖╕Ά▓ΔΆ│╝ Ά░βΈΜνΆ│ι Έ│┤Έσ┤ ΈΡεΈΜν.
export default function* root() {
  yield spawn(watchBoard);
}

Ήβε watcherΈΥνΉζΕ spawnΉε╝Έκε Έυ╢ΉΨ┤ΉΧ╝ΊΧαΈΛΦΉπΑΈΛΦ Ήζ┤ΉΛΙΉΩΡ ΈΓαΉβΑΉηΙΈΜν. (ΉΩυΆ╕░ΉΩΦ ΉηΡΈΠβΉε╝Έκε sagaΆ░Α ΉηυΉΜεΉηΣΈΡαΈΛΦ ΊΝρΊΕ┤ΈΠΕ ΉηΙΈΛΦΈΞ░, ΉΧΕΉπΒ ΊβεΉγσΊΧ┤Έ│╕ ΉιΒΉζΑ ΉΩΗΈΜν)

action => saga => action => reducer Έκε ΉΩ░Ά▓░ΈΡαΈΛΦ sagaΆ░Α ΉβΕΉΕ▒ΈΡαΉΩΙΈΜν.

store/index.js
import createSagaMiddleware, { END } from "redux-saga";
const saga = createSagaMiddleware();

export default function configureStore(initialState) {
  const store = createStore(
    rootReducer,
    initialState,
    compose(
      applyMiddleware(...middlewares),
      inDevelopment && window.devToolsExtension
        ? window.devToolsExtension()
        : (f) => f
    )
  );

  store.runSaga = saga.run;
  store.close = () => store.dispatch(END);

  return store;
}
