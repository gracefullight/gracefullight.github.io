<?php
public function sendResetLinkEmail(Request $request) {
  $this->validateId($request);
  // emailмқҙ м•„лӢҢ м•„мқҙл””лЎң кІҖмғүмқ„ н•ҙ л©”мқјмқ„ м°ҫкі  мӢ¶лӢӨл©ҙ
  $data = Member::where('id', $request->only('id'))
    ->first(['email']);

  $email = isset($data->email) ? $data->email : null;

  // м—¬кё°м„ң emailмқҙ User Model(Member Model)м—җ л°”мқёл”©лҗңлӢӨ.
  // м•„лһҳмқҳ sendPasswordResetNotification л©”мҶҢл“ңм—җм„ң $this->email к°’мқҙлӢӨ.
  $response = $this->broker()->sendResetLink([
    'email' => $email
  ]);

  // лҰ¬н„ҙ к°’мқ„ ліҖкІҪн•ҙмЈјмһҗ.
  return response()->json([
    'email' => $email
  ], $response == Password::RESET_LINK_SENT ? 200 : 500 );
}

public function auth()
{

  $this->post('login', 'Auth\LoginController@login');
  $this->post('logout', 'Auth\LoginController@logout')->name('logout');

  // Registration Routes...

  $this->post('password/reset', 'Auth\ResetPasswordController@reset');
}

AuthвҖӢ

Basic AuthлҘј мӮ¬мҡ©н•ҳлҠ”лҚ° кұҙл“ңл Өм•ј лҗҳлҠ” кіімқҙ л§ҺмңјлҜҖлЎң Auth лӘЁл“Ҳл§Ң мӮ¬мҡ©н•ҳлҠ”кІҢ м •мӢ кұҙк°•м—җ мўӢлӢӨкі  ліёлӢӨ. (Bootstrap кё°л°ҳмқҳ Laravelм—җ л”ұ л§һлҠ” лӘЁм–‘мқ„ мһ…мқҖ н”„лЎңм қнҠёлқјл©ҙ кё°ліё мқёмҰқмқҙ мўӢкІ м§Җл§Ң)

лЎңк·ёмқёвҖӢ

лЁјм Җ мӮ¬мҡ©н•  лӘЁлҚём—җ Authenticatable нҒҙлһҳмҠӨлҘј мғҒмҶҚ л°ӣмһҗ к·ёлҰ¬кі  LoginControllerм—җм„ң Auth::attempt() л©”мҶҢл“ңлҘј мӢӨн–үн•ҳл©ҙ лҒқмқҙлӢӨ.

app/Http/Controllers/LoginController.php
public function login(Request $request) {
  $password = $request->password;

  if(Auth::attempt([ 'id' => $request->id, 'password' => $password ])){
    return response(null, 200);

  }else{
    return abort(401);
  }
}

not bcryptвҖӢ

Auth лӘЁл“ҲмқҖ кё°ліёмңјлЎң bcryptлҘј мӮ¬мҡ©н•ҙ л№„л°ҖлІҲнҳёлҘј м•”нҳёнҷ”н•ҳкі  л№„көҗн•ҳлҠ”лҚ° лӢӨлҘё м•”нҳёнҷ” л°©мӢқмқ„ мӮ¬мҡ©н•ҙм•јн•ҳлҠ” кІҪмҡ°к°Җ мһҲлӢӨ. bcryptлҘј мӮ¬мҡ©н•ҳм§Җ м•ҠкІҢ мІҳлҰ¬н•ҙліҙмһҗ.

app/Models/Member.php
// мқҙ л©”мҶҢл“ңлҘј override н•ҙм•јн•ңлӢӨ.
public function getAuthPassword() {
  // bcrypt л№„көҗлҘј н•ҳм§Җ м•Ҡкё° мң„н•ҙ к°•м ңлЎң н•ҙмӢңлҘј мғқм„ұн•ңлӢӨ.
  return Hash::make($this->password);

мқҙм ң Auth::attempt() л©”мҶҢл“ңм—җ нҢЁмҠӨмӣҢл“ңлҘј л„ҳкёё л•Ң м•”нҳёнҷ”лҘј н•ҙмЈјкі  л„ҳкё°л©ҙ лҗңлӢӨ.

лҜёл“ӨмӣЁм–ҙвҖӢ

кҙҖлҰ¬мһҗ м„ём…ҳмқҙ мқёмҰқлҗң мӮ¬лһҢл§Ң кҙҖлҰ¬мһҗ нҺҳмқҙм§Җм—җ м ‘к·јн•  мҲҳ мһҲм–ҙм•јн•ңлӢӨ. php artisan make:middleware Admin лӘ…л №м–ҙлҘј мӢӨн–үн•ҙ Admin MiddlewareлҘј л§Ңл“Өмһҗ.

app/Http/Middleware/Admin.php
<?php
use Auth;
use Closure;

class Admin {

  public function handle($request, Closure $next) {
    // admin sessionмқҙ мһҲлҠ” кІҪмҡ°л§Ң requestлҘј м§„н–үн•ңлӢӨ.
    if (Auth::guard('admin')->check()) {
      return $next($request);
    }

    // м•„лӢҗкІҪмҡ° кҙҖлҰ¬мһҗ лЎңк·ёмқё нҺҳмқҙм§ҖлЎң л„ҳкёҙлӢӨ.
    return redirect()->route('admin.login');
  }
}
