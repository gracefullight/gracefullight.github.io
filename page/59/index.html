import crypto from "node:crypto";

class TripleDes {
  // #iv;
  // #key;

  constructor(key, iv) {
    this.key = key;
    this.iv = iv;
  }

  getKey() {
    return this.key.padEnd(24, String.fromCharCode(0));
  }

  getIv() {
    return this.iv;
  }

  encrypt(plain, iv) {
    if (!iv) {
      iv = this.iv ? Buffer.from(this.iv, "hex") : crypto.randomBytes(8);
    }

    const cipher3des = crypto.createCipheriv(
      "des-ede3-cfb8",
      this.getKey(),
      iv,
    );
    let encrypted = cipher3des.update(plain, "utf8", "hex");
    encrypted += cipher3des.final("hex");

    this.iv = iv.toString("hex");
    return encrypted;
  }

  decrypt(encrypted, iv) {
    if (!iv) {
      iv = this.iv;
    }

    const decipher3des = crypto.createDecipheriv(
      "des-ede3-cfb8",
      this.getKey(),
      Buffer.from(iv, "hex"),
    );
    let decrypted = decipher3des.update(encrypted, "hex", "utf8");
    decrypted += decipher3des.final("utf8");

    return decrypted;
  }
}

module.exports = TripleDes;

нҢЁн„ҙвҖӢ

  • мҳөм ёлІ„лё”мқҳ ліҖмҲҳ мҠӨнғҖмқјмқҖ мқҙлҰ„ л’Өм—җ const click$ мІҳлҹј $мқ„ л’Өм—җ л¶ҷнҳҖмЈјлҠ”кІҢ м •нҳ•нҷ”лҗҳм–ҙмһҲлӢӨ.
  • subscribe лҠ” next, error, completeлҘј нҢҢлқјлҜён„°лЎң л°ӣлҠ”лӢӨ.

л§Ҳлё” лӢӨмқҙм–ҙк·ёлһЁвҖӢ

л§Ҳлё” лӢӨмқҙм–ҙк·ёлһЁмқҖ м—°мӮ°мһҗлҘј мүҪкІҢ мқҙн•ҙн•ҳкё° мң„н•ҙ, мҳөм ёлІ„лё”мқ„ н…ҢмҠӨнҠён•ҳкё°мң„н•ҙ лҸ„мӢқнҷ”лҗң лӢӨмқҙм–ҙк·ёлһЁмқҙлӢӨ. нғҖмқҙл°Қкіј к°’мқҳ ліҖнҷ”лҘј н•ң лҲҲм—җ нҢҢм•…н•  мҲҳ мһҲлӢӨ.

  • мң„м—җ мӨ„мқҖ input мқҙкі  м—°мӮ°мһҗлҘј л§ҢлӮҳкі лӮҳл©ҙ м•„лһҳ мӨ„м—җм„ң output мқҙ м–ҙл–»кІҢ л°”лҖҢлҠ”м§Җ ліҙм—¬м§„лӢӨ.
  • к°ҖлЎң мӨ„мқҖ н•ҳлӮҳмқҳ мҳөм ёлІ„лё”мқҙлӢӨ.
  • мӨ„мқҳ | нҢҢмқҙн”„лҠ” кө¬лҸ… мҷ„лЈҢлҘј лӮҳнғҖлӮёлӢӨ.

мғқм„ұвҖӢ

кө¬лҸ…н•ҙм ңвҖӢ

мҳөм ёлІ„лё”мқҳ лҰ¬н„ҙн•ЁмҲҳлЎң кө¬лҸ…н•ҙм ң мҪңл°ұмқ„ м§Җмӣҗн•ңлӢӨ, мҪңл°ұмқҙ н•„мҡ”н•ҳм§Җ м•ҠлӢӨл©ҙ unsubscribe() лҘј нҳём¶ңн•ҙмЈјкё°л§Ң н•ҳл©ҙ лҗңлӢӨ.

const test$ = Observable.create((observer) => {
  const interval = setInterval(() => {
    console.log("test");
  }, 1000);

  return () => {
    clearInterval(interval);
  };
});

const subscription = test$.subscribe();
subscription.unsubscribe();

нҢҢмқҙнҚјлё” м—°мӮ°мһҗвҖӢ

Pipeable м—°мӮ°мһҗлҠ” мҳөм ёлІ„лё” мқёмҠӨн„ҙмҠӨлҘј pipe н•ЁмҲҳ м•Ҳм—җм„ң лӢӨлЈ° мҲҳ мһҲлҠ” м—°мӮ°мһҗмқҙлӢӨ. кё°ліём ҒмңјлЎң rxjs/operators лқјмқҙлёҢлҹ¬лҰ¬ м•Ҳм—җ л“Өм–ҙмһҲлӢӨ.

const { map } = require("rxjs/operators");

const test$ = Observable.create((observer) => {
  observer.next(1);
  observer.next(2);
  observer.complete();
});

test$.pipe(map((value) => value * 2)).subscribe((item) => console.log(item));

/*
2
4
*/

fromEventвҖӢ

EventEmitter нҒҙлһҳмҠӨмқҳ к°қмІҙмҷҖ мЎ°н•©н•ҳкұ°лӮҳ лёҢлқјмҡ°м Җмқҳ мқҙлІӨнҠёлҘј мҳөм ёлІ„лё”лЎң л°”кҝҖ л•Ң мӮ¬мҡ©н•ңлӢӨ.

deferвҖӢ

нҢ©нҶ лҰ¬ н•ЁмҲҳлЎң мҳөм ёлІ„лё”мқ„ мғқм„ұн•ң нӣ„ кө¬лҸ…н•ңлҠҗ мӢңм җм—җ нҢ©нҶ лҰ¬ н•ЁмҲҳлҘј нҳём¶ңн•ҙ мқҙлҜё мғқм„ұн•ң мҳөм ёлІ„лё”мқ„ лҰ¬н„ҙл°ӣм•„ кө¬лҸ…н•ңлӢӨ.

from кіјмқҳ м°ЁмқҙлҠ” лӢӨлҰ„кіј к°ҷлӢӨ.

  • from
    • н”„лЎңлҜёмҠӨ лӮҙл¶Җ кө¬нҳ„л¶Җк°Җ м–ём ң мӢӨн–үлҗҳлҚҳм§Җ мғҒкҙҖ м—Ҷмқ„ л•Ң
    • мқҙлҜё мӢӨн–ү мӨ‘мқҙкұ°лӮҳ мҷ„лЈҢн•ң н”„лЎңлҜёмҠӨлҘј мҳөм ёлІ„лё”лЎң л§Ңл“Ө л•Ң
  • defer
    • мҳөм ёлІ„лё”мқ„ кө¬лҸ…н•ҳлҠ” мӢңм җм—җ н”„лЎңлҜёмҠӨлҘј мғқм„ұн•ҙ кө¬нҳ„л¶Җк°Җ мӢӨн–үлҗҳм–ҙм•јн•  л•Ң
    • н”„лЎңлҜёмҠӨ к°қмІҙ мғқм„ұ мӢңм җмқҙ кө¬лҸ…н•ҳлҠ” мӢңм җмқҙм—¬м•јн•  л•Ң

rangeвҖӢ

лІ”мң„ м§Җм • нӣ„ к·ё к°’мқ„ мҲңм„ңлҢҖлЎң л°ңн–үн•ңлӢӨ. л°ҳліөл¬ёмқҙ н•„мҡ”н•  л•Ң мӮ¬мҡ©лҗңлӢӨ.

intervalвҖӢ

ms лӢЁмң„лЎң к°’мқ„ л°ңн–үн•ңлӢӨ.

timerвҖӢ

нҢҢлқјлҜён„°к°Җ н•ҳлӮҳмқј кІҪмҡ° ms мқҙнӣ„м—җ н•ң лІҲ к°’мқ„ л°ңн–үн•ҳкі , л‘җ к°ңмқј кІҪмҡ° ms мқҙнӣ„м—җ л‘җлІҲ м§ё нҢҢлқјлҜён„°л§ҢнҒј мЈјкё°м ҒмңјлЎң к°’мқ„ л°ңн–үн•ңлӢӨ.

emptyвҖӢ

к°’ л°ңн–ү нӣ„ мӨ‘к°„м—җ л©Ҳм¶°м•јн•ҳлҠ” мғҒнҷ©м—җ мӮ¬мҡ©н•ңлӢӨ. мқҙ н•ЁмҲҳл§Ң мӮ¬мҡ©н•ҳм§ҖлҠ” м•Ҡкі  лӢӨлҘё н•ЁмҲҳлӮҳ м—°мӮ°мһҗмҷҖ мЎ°н•©н•ҙм„ң complete н•ЁмҲҳлҘј нҳём¶ңн•ҙм•ј н•  л•Ң мӮ¬мҡ©лҗңлӢӨ. мҰү, л°”лЎң кө¬лҸ…мқ„ мҷ„лЈҢн•ҙм•јлҗ л•Ң мӮ¬мҡ©лҗңлӢӨ.

throwErrorвҖӢ

filterвҖӢ

мЈјлЎң нҢҢмқҙнҚјлё” м—°мӮ°мһҗмҷҖ м—°кІ°н•ҙм„ң мӮ¬мҡ©лҗңлӢӨ.

lastвҖӢ

takeUntilвҖӢ

нҠ№м • мқҙлІӨнҠёк°Җ л°ңмғқн•  л•Ңк№Ңм§Җ мҳөм ёлІ„лё”мқ„ кө¬лҸ…н•ҙм•јн•  л•Ң мӮ¬мҡ©н•ңлӢӨ. мҳҲмӢңлЎң ліҙлҠ” кІҢ л№ лҘҙлӢӨ.

  .pipe(
