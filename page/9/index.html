 *
 * ьЭ┤ ьШИьаЬьЧРьДЬыКФ ьвЕьЖНьД▒ ъ╖╕ыЮШэФДые╝ ызМыУдъ│а ьЭ┤ ъ╖╕ыЮШэФДые╝ ьВмьЪйэХШьЧм ыкиыУа ыкиыУИыУдьЭД эХШыВШьЭШ ы▓ИыУдыбЬ эМиэВдьзХ эХйыЛИыЛд.
 * ъ╖╕ыЯ╝ ьЛЬьЮСэХ┤ ы│┤ъ▓аьК╡ыЛИыЛд :)
 *
 * ь░╕ъ│а: ьЭ┤ ьШИьаЬыКФ ыздьЪ░ ыЛиьИЬэЩФыРШьЦ┤ ьЮИьК╡ыЛИыЛд. ьИЬэЩШ ь░╕ьб░, ь║РьЛ▒ ыкиыУИ, эММьЛ▒ ь╡ЬьаБэЩФ ыУ▒ьЧР ыМАэХЬ ыВ┤ьЪйьЭА ьГЭыЮ╡
 * эХШьЧм ъ░АыКеэХЬъ░А ыЛиьИЬэХШъ▓М ызМыУдьЧИьК╡ыЛИыЛд.
 */

const fs = require('fs');
const path = require('path');
const babylon = require('babylon');
const traverse = require('babel-traverse').default;
const { transformFromAst } = require('babel-core');

let ID = 0;

// ьЪ░ьДа file pathые╝ ы░ЫыКФ эХиьИШые╝ ьГЭьД▒эХШъ│а
// эММьЭ╝ьЭД ыВ┤ьЪйьЭД ьЭ╜ъ│а, ьвЕьЖНьД▒ьЭД ь╢Фь╢ЬэХйыЛИыЛд.
function createAsset(filename) {
  // эММьЭ╝ьЭШ ыВ┤ьЪйьЭД ым╕ьЮРьЧ┤ыбЬ ьЭ╜ьК╡ыЛИыЛд.
  const content = fs.readFileSync(filename, 'utf-8');

  // ьЭ┤ьаЬ ьЭ┤ эММьЭ╝ьЭ┤ ьЦ┤ыЦд эММьЭ╝ьЧР ьвЕьЖНыРШыКФьзА ьХМьХДы│┤ъ▓аьК╡ыЛИыЛд. ьЪ░ыжмыКФ import ым╕ьЮРьЧ┤ьЭД ы│┤ъ│а ьЭШьб┤ьД▒ьЭД
  // эММьХЕэХа ьИШ ьЮИьК╡ыЛИыЛд эХШьзАызМ, ьЭ┤ъ▓ГьЭА ыЛиьИЬэХЬ ьаСъ╖╝ы▓ХьЭ┤ьЦ┤ьДЬ, ыМАьЛаьЧР ьЮРы░ФьКдэБмыж╜эК╕ эММьДЬые╝ ьВмьЪйэХШъ▓аьК╡ыЛИыЛд.

  // ьЮРы░ФьКдэБмыж╜эК╕ эММьДЬыУдьЭА ьЮРы░ФьКдэБмыж╜эК╕ ь╜ФыУЬые╝ ьЭ╜ъ│а ьЭ┤эХ┤эХа ьИШ ьЮИыПДыбЭ ыПДьЩАьг╝ыКФ эИ┤ьЮЕыЛИыЛд.
  // эММьДЬыКФ AST(abstract syntax tree)ыЭ╝ыКФ ьвАыНФ ь╢ФьГБэЩФыРЬ ыкиыН╕ьЭД ьГЭьД▒эХйыЛИыЛд.
  //
  // ASTьЧР ыМАэХ┤ ьЭ┤эХ┤эХШыадый┤ AST Explorer(https://astexplorer.net)ьЭД ъ╝н ы│┤ъ╕░ые╝ ъ░ХыаеэХШъ▓М ь╢Фь▓ЬэХйыЛИыЛд.
  // ASTъ░А ьЦ┤ыЦ╗ъ▓М ьЭ┤ыгиьЦ┤ьа╕ ьЮИыКФьзА эЩХьЭ╕эХа ьИШ ьЮИьК╡ыЛИыЛд.
  //
  // ASTыКФ ьЪ░ыжмьЭШ ь╜ФыУЬьЧР ыМАэХ┤ ызОьЭА ьаХы│┤ые╝ ъ░АьзАъ│а ьЮИьК╡ыЛИыЛд. ьЪ░ыжмыКФ ь┐╝ыжмые╝ ьЭ┤ьЪйэХШьЧм
  // ьЪ░ыжмьЭШ ь╜ФыУЬъ░А эХШыадыКФ ьЭ╝ьЧР ыМАэХ┤ ьЭ┤эХ┤эХа ьИШ ьЮИьК╡ыЛИыЛд.
  const ast = babylon.parse(content, {
    sourceType: 'module',
  });

  // ьЭ┤ ы░░ьЧ┤ьЭА эШДьЮм ыкиыУИьЭШ ьЭШьб┤ьД▒ьЭД ьГБыМА ъ▓╜ыбЬыбЬ ъ░АьзАъ│а ьЮИьЭД ъ▓ГьЮЕыЛИыЛд.
  const dependencies = [];

  // ьЪ░ыжмыКФ AST ьИЬэЪМые╝ эЖ╡эХ┤ ъ░Бъ░БьЭШ ыкиыУИыУдьЭ┤ ьЦ┤ыЦд ьЭШьб┤ьД▒ьЭД ъ░АьзАъ│а ьЮИыКФьзА ьЭ┤эХ┤эХШыад эХйыЛИыЛд.
  // ьЭ┤ъ▓ГьЭД эЖ╡эХ┤ ASTьХИьЧРьДЬ ыкиыУа import keyword ьДаьЦ╕ьЭД эММьХЕэХа ьИШ ьЮИьК╡ыЛИыЛд.
  traverse(ast, {
    // ECMAScript ыкиыУИыУдьЭА ьаХьаБьЭ┤ыпАыбЬ ыздьЪ░ эММьХЕэХШъ╕░ ьЙ╜ьК╡ыЛИыЛд.ьЭ┤ыКФ ы│АьИШые╝ ъ░Аьа╕ьШм ьИШ ьЧЖъ▒░ыВШ ьб░ъ▒┤ы╢АыбЬ
    // ыЛдые╕ ыкиыУИьЭД ъ░Аьа╕ьШм ьИШ ьЧЖьЭМьЭД ьЭШып╕эХйыЛИыЛд. import ъ╡мы╢ДьЭД ы│╝ ыХМ ызИыЛд ь╣┤ьЪ┤эМЕьЭД эХШъ│а ьЭШьб┤ьД▒ьЭД ъ░АьзАъ│а
    // ьЮИыКФ ъ▓ГьЬ╝ыбЬ ъ░Дьг╝ эХа ьИШ ьЮИьК╡ыЛИыЛд.
    ImportDeclaration: ({ node }) => {
      // import ъ╡мым╕ызИыЛд dependencies ы░░ьЧ┤ьЧР ъ░ТьЭД ь╢Фъ░АэХйыЛИыЛд.
      dependencies.push(node.source.value);
    },
  });

  // ыШРэХЬ ъ░ДыЛиэХЬ ь╣┤ьЪ┤эД░ые╝ ьЭ┤ьЪйэХШьЧм ьЭ┤ ыкиыУИьЧР ъ│аьЬа ьЛЭы│ДьЮРые╝ эХаыЛ╣эХйыЛИыЛд.
  const id = ID++;

  // ьЪ░ыжмыКФ ьЭ╝ы╢А ы╕МыЭ╝ьЪ░ьаАьЧРьДЬызМ ьзАьЫРэХШыКФ ECMAScript moduleыУдьЭ┤ыВШ ъ╕░ыКеыУдьЭД ьВмьЪйэХа ъ░АыКеьД▒ыПД ьЮИьК╡ыЛИыЛд.
  // ьЪ░ыжмъ░А ызМыУЬыКФ ы▓ИыУдьЭ┤ ыкиыУа ы╕МыЭ╝ьЪ░ьаАьЧРьДЬ ыПМьХДъ░АыПДыбЭ BabelьЭД ьЭ┤ьЪйэХ┤ьДЬ transpileэХа ьИШ ьЮИьК╡ыЛИыЛд
  // (https://babeljs.io ь░╕ъ│а).
  //
  // `presets` ьШ╡ьЕШьЭА BabelьЭ┤ ьЦ┤ыЦ╗ъ▓М ьЪ░ыжм ь╜ФыУЬые╝ ы░Фъ┐АьзАьЧР ыМАэХ┤ ъ▓░ьаХэХйыЛИыЛд. ьЪ░ыжмыКФ `babel-preset-env`
  // ые╝ ьЭ┤ьЪйэХШьЧм ыМАы╢Аы╢ДьЭШ ы╕МыЭ╝ьЪ░ьаАьЧРьДЬ ьЪ░ыжмьЭШ ь╜ФыУЬые╝ ьВмьЪйэХа ьИШ ьЮИыПДыбЭ ы░Фъ╛╕ыПДыбЭ эХШъ▓аьК╡ыЛИыЛд.
  const { code } = transformFromAst(ast, null, {
    presets: ['env'],
  });

  // ьЭ┤ ыкиыУИьЧР ыМАэХЬ ьаХы│┤ые╝ return эХйыЛИыЛд.
  return {
    id,
    filename,
    dependencies,
    code,
  };
}

// ьЭ┤ьаЬ ыЛиьЭ╝ ыкиыУИьЭШ ьвЕьЖНьД▒ьЭД ь╢Фь╢ЬэХа ьИШ ьЮИьЬ╝ыпАыбЬ, entry fileьЭШ ьЭШьб┤ьД▒ьЭД ь╢Фь╢ЬэХШыКФ ъ▓Гы╢АэД░ ьЛЬьЮСэХШъ▓аьК╡ыЛИыЛд.
// ьЭ┤ ьЮСьЧЕьЭА ьХаэФМыжмь╝АьЭ┤ьЕШьЭШ ыкиыУа ыкиыУИъ│╝ ъ░Б ыкиыУИьЭ┤ ьДЬыбЬ ьЦ┤ыЦ╗ъ▓М ьЭШьб┤эХШыКФьзАые╝ эММьХЕэХа ыХМъ╣МьзА ъ│ДьЖН ьзДэЦЙэХа ъ▓ГьЮЕыЛИыЛд.
// ьЭ┤ ьЮСьЧЕьЭШ ъ▓░ъ│╝ым╝ьЭД ьЭШьб┤ьД▒ ъ╖╕ыЮШэФДыЭ╝ ы╢АыжЕыЛИыЛд.
function createGraph(entry) {
  // entry fileы╢АэД░ ы╢ДьДЭьЭД ьЛЬьЮСэХйыЛИыЛд.
  const mainAsset = createAsset(entry);

  // queueые╝ ьВмьЪйэХ┤ьДЬ ыкиыУа assetьЭШ ьЭШьб┤ьД▒ьЭД ы╢ДьДЭэХШыПДыбЭ эХШъ▓аьК╡ыЛИыЛд. ьЭ┤ ьЮСьЧЕьЭД ьЬДэХ┤
  // entry assetьЭД ъ░АьзАъ│а ьЮИыКФ ы░░ьЧ┤ьЭД ьаХьЭШэХйыЛИыЛд.
  const queue = [mainAsset];

  // ьЧмъ╕░ьДЬ queueьЭШ ы░Шы│╡ьЭД ьЬДэХ┤ `for ... of` ы░Шы│╡ым╕ьЭД ьВмьЪйэХйыЛИыЛд. ь▓ШьЭМьЧРыКФ queueъ░А assetьЭД эХШыВШызМ
  // ъ░АьзАъ│а ьЮИьзАызМ ьЮСьЧЕьЭ┤ ы░Шы│╡ыРШыКФ ыПЩьХИьЧР ьГИыбЬьЪ┤ assetыУдьЭД queueьЧР ь╢Фъ░АэХйыЛИыЛд. ьЭ┤ ы░Шы│╡ым╕ьЭА queueъ░А
  // ы╣ДьЦ┤ьзИ ыХМ ъ╣МьзА ъ│ДьЖНыРйыЛИыЛд.
  for (const asset of queue) {
    // ыкиыУа assetыУдьЭА ьЭШьб┤ьД▒ьЭ┤ ьЮИыКФ ыкиыУИьЧР ыМАэХЬ ьГБыМАъ▓╜ыбЬыУдьЭД ыжмьКдэК╕ыбЬ ъ░АьзАъ│а ьЮИьК╡ыЛИыЛд. ьЪ░ыжмыКФ ъ╖╕ ыжмьКдэК╕ые╝
    // ьИЬэЪМэХШый┤ьДЬ `createAsset()`эХиьИШыбЬ ы╢ДьДЭэХШъ│а, ьХДыЮШ ъ░Эь▓┤ые╝ эЖ╡эХШьЧм ыкиыУИыУдьЭШ ьЭШьб┤ьД▒ьЭД ь╢Фь▓ЩэХа ъ▓ГьЮЕыЛИыЛд.
    asset.mapping = {};

    // ьЭ┤ъ▓ГьЭА ьЭ┤ ыкиыУИьЭ┤ ьЮИыКФ ыФФыаЙэЖаыжмьЮЕыЛИыЛд.
    const dirname = path.dirname(asset.filename);

    // ьвЕьЖНьД▒ьЧР ыМАэХЬ ьГБыМА ъ▓╜ыбЬ ыжмьКдэК╕ые╝ ьИЬэЪМэХйыЛИыЛд.
    asset.dependencies.forEach((relativePath) => {
      // `createAsset()` эХиьИШыКФ ьаИыМА ъ▓╜ыбЬъ░А эХДьЪФэХйыЛИыЛд. dependencies ы░░ьЧ┤ьЭА ьГБыМА ъ▓╜ыбЬые╝ ъ░АьзАъ│а
      // ьЮИыКФ ы░░ьЧ┤ьЮЕыЛИыЛд. ьЭ┤ыЯмэХЬ ъ▓╜ыбЬыУдьЭА ыкиыУИьЭ┤ importыРЬ fileьЧР ыФ░ыЭ╝ ьГБыМАьаБьЮЕыЛИыЛд. ыФ░ыЭ╝ьДЬ ы╢Аыки assetьЭШ
      // ъ▓╜ыбЬые╝ ьЭ┤ьЪйэХ┤ьДЬ ьГБыМА ъ▓╜ыбЬые╝ ьаИыМАъ▓╜ыбЬыбЬ ы░Фъ┐ФьХ╝ эХйыЛИыЛд.
      const absolutePath = path.join(dirname, relativePath);

      // assetьЭШ ыВ┤ьЪйьЪ╕ ы╢ДьДЭэХШъ│а, ыВ┤ьЪйьЭД ьЭ╜ъ│а, ьЭШьб┤ьД▒ьЭД ь╢Фь╢ЬэХйыЛИыЛд.
      const child = createAsset(absolutePath);

      // `asset`ьЭШ ьЭШьб┤ьД▒ьЭА `child`ьЧРъ▓М ыЛмыадьЮИьК╡ыЛИыЛд. ьЪ░ыжмыКФ `mapping` ъ░Эь▓┤ьЧР relativePathьЩА child.idые╝
      // ьЭ┤ьЪйэХ┤ьДЬ ъ┤Аъ│Дые╝ эСЬэШДэХа ьИШ ьЮИьК╡ыЛИыЛд.
      asset.mapping[relativePath] = child.id;

      // ызИьзАызЙьЬ╝ыбЬ child assetьЭД queueьЧР ь╢Фъ░АэХШьЧм ъ╡мым╕ ы╢ДьДЭьЭ┤ ы░Шы│╡ыРШыПДыбЭ эХйыЛИыЛд.
      queue.push(child);
    });
  }

  // ьЭ┤ ьЛЬьаРьЧРьДЬ queueыКФ ьХаэФМыжмь╝АьЭ┤ьЕШьЭШ ыкиыУа ыкиыУИьЭ┤ эПмэХиыРЬ ы░░ьЧ┤ьЮЕыЛИыЛд. ьЭ┤ъ▓ГьЭ┤ ьЪ░ыжмъ░А ъ╖╕ыЮШэФДые╝ эСЬэШДэХШыКФ ы░йы▓ХьЮЕыЛИыЛд.
  return queue;
}

// ыЛдьЭМьЬ╝ыбЬ, ъ╖╕ыЮШэФДые╝ ьЭ┤ьЪйэХШьЧм ы╕МыЭ╝ьЪ░ьаАьЧРьДЬ ьЛдэЦЙэХа ьИШ ьЮИыКФ ы▓ИыУдьЭД ы░ШэЩШэХШыКФ эХиьИШые╝ ьаХьЭШэХйыЛИыЛд.
//
// ьЪ░ыжмьЭШ ы▓ИыУдьЭА self-invoking(ьЮРьЛаьЭД ы╢Аые╝ьИШ ьЮИыКФ)эХиьИШые╝ ъ░АьзАъ│а ьЮИьК╡ыЛИыЛд.
//
// (function() {})()
//
// ьЭ┤ эХиьИШыКФ эХШыВШьЭШ ьЭ╕ьЮРызМ ы░ЫьЭД ьИШ ьЮИьК╡ыЛИыЛд: ыкиыУа ыкиыУИьЭШ ьаХы│┤ые╝ ъ░АьзАъ│а ьЮИыКФ ъ╖╕ыЮШэФД.
function bundle(graph) {
  let modules = '';

  // ьЭ┤ эХиьИШые╝ ъ╡мьД▒эХШъ╕░ ьаДьЧР ыздъ░Ь ы│АьИШыбЬ ьаДыЛмэХа ъ░Эь▓┤ые╝ ызМыУдъ▓аьК╡ыЛИыЛд. ы░ШыУЬьЛЬ ьХМьХДыСмьХ╝эХа ъ▓ГьЭА ьЪ░ыжмъ░А ызМыУЬыКФ
  // ьКдэК╕ызБьЭА 2ъ░ЬьЭШ ьдСъ┤ДэШ╕({})ыбЬ ъ░РьЛ╕ьа╕ ьЮИьЦ┤ьХ╝ эХЬыЛдыКФ ъ▓ГьЮЕыЛИыЛд. ьЪ░ыжмыКФ ыЛдьЭМъ│╝ ъ░ЩьЭА эПмыйзьЬ╝ыбЬ ь╢Фъ░АэХа
  // ъ▓ГьЮЕыЛИыЛд: `key: value,`.
  graph.forEach((mod) => {
    // ъ╖╕ыЮШэФДьХИьЧР ьЮИыКФ ыкиыУа ыкиыУИыУдьЭА entryые╝ ъ░Эь▓┤ыбЬ ъ░АьзАъ│а ьЮИьК╡ыЛИыЛд. ьЪ░ыжмыКФ moduleьЭШ idые╝
    // ъ░ТьЧР ыМАэХЬ эВдыбЬ ьВмьЪйэХйыЛИыЛд.(ъ░Б ыкиыУИызИыЛд 2ъ░ЬьЭШ ъ░ТьЭ┤ ьЮИьК╡ыЛИыЛд.)
    //
    // ь░╗ы▓Иьз╕ ъ░ТьЭА эХиьИШыбЬ ъ░РьЛ╝ ъ░Б ыкиыУИьЭШ ь╜ФыУЬьЮЕыЛИыЛд. ъ╖╕ ьЭ┤ьЬаыКФ ыкиыУИьЭШ scopeые╝ ьзАьаХэХ┤ьХ╝ эХШъ╕░ ыХМым╕ьЮЕыЛИыЛд.
    // эХЬ ыкиыУИьЧРьДЬ ы│АьИШые╝ ьаХьЭШэХШый┤ ыЛдые╕ ыкиыУИьЭ┤ыВШ ъ╕АыбЬы▓М scopeьЧР ьШБэЦеьЭД ьг╝ьзА ьХКьХДьХ╝ эХйыЛИыЛд.
    //
    // transpiledыРЬ ыкиыУИыУдьЭА CommonJS ыкиыУИ ьЛЬьКдэЕЬьЭД ьВмьЪйэХйыЛИыЛд:
    // эХ┤ыЛ╣ ыкиыУИ ьЛЬьКдэЕЬьЭА `require`, `module`, ъ╖╕ыжмъ│а `exports`ые╝ эЖ╡эХ┤ ыкиыУИэЩФ эХйыЛИыЛд.
