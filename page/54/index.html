// IndexedDB лҠ” Promisify лҗҳм–ҙмһҲм§Җ м•Ҡм•„м„ң лқјмқҙлёҢлҹ¬лҰ¬к°Җ н•„мҡ”н•ҳлӢӨ.
importScripts(
  "https://cdn.jsdelivr.net/npm/localforage@1.7.3/dist/localforage.min.js",
);

// мәҗмӢңн•ҳкі  мӢ¶мқҖ POST м—”л“ңнҸ¬мқёнҠё
const ENDPOINT = "https://your-domain/post-request";

const bin2Hex = (buffer) => {
  let digest = "";
  const dataView = new DataView(buffer);
  for (let i = 0, len = dataView.byteLength; i < len; i += 4) {
    const value = dataView.getUint32(i);
    // hex лЎң л°”кҫёл©ҙ нҢЁл”©л№„нҠё 0 мқҙ м ңкұ°лҗңлӢӨ.
    const hex = value.toString(16);
    // uint32 лҠ” 4bytes лЎң лӮҳмҳЁлӢӨ.
    const padding = "00000000";
    // нҢЁл”©мқ„ лҚ”н•ҙм„ң л’Өм—җм„ң мһҳлқјмӨҖлӢӨ.
    const paddedValue = (padding + hex).slice(-padding.length);
    digest += paddedValue;
  }

  return digest;
};

const postRequestFetchListener = (fetchEvent) => {
  const requestUrl = fetchEvent.request.url;
  const method = fetchEvent.request.method.toUpperCase();
  // л§һлҠ” м—”л“ңнҸ¬мқёнҠёмқём§Җ нҷ•мқё
  if (!(method === "POST" && requestUrl === ENDPOINT)) {
    return;
  }

  fetchEvent.respondWith(
    fetchEvent.request
      .clone()
      .arrayBuffer()
      .then((buffer) => {
        const requestBody = String.fromCharCode.apply(
          null,
          new Uint8Array(buffer),
        );
        // request body м—җ мӣҗн•ҳлҠ” мЎ°кұҙл§Ң мәҗмӢңмІҳлҰ¬н•  мҲҳ мһҲкІҢ н•ңлӢӨ.
        if (requestBody.includes("cache=1")) {
          // мҶҚлҸ„л©ҙм—җм„ң лӢӨлҘё н•ҙмӢұ м•Ңкі лҰ¬мҰҳмқ„ мӮ¬мҡ©н•ҙлҸ„ л¬ҙл°©н•ҳлӢӨ.
          return crypto.subtle.digest("SHA-1", buffer);
        }

        return Promise.reject();
      })
      .then((sha1Buffer) => {
        const sha1Hash = bin2Hex(sha1Buffer);
        console.log("SHA1 Hash => ", sha1Hash);

        // IndexedDB м—җм„ң мәҗмӢңлҗң нӮӨлҘј м°ҫлҠ”лӢӨ.
        return localforage.getItem(sha1Hash).then((cachedResponse) => {
          if (cachedResponse) {
            console.log("Cached repsonse => ", cachedResponse);
            return new Response(cachedResponse, {
              // м—¬кё°м„ң statusCode лҘј 304 лЎң лӮҙліҙлӮҙкі  мӢ¶м—ҲмңјлӮҳ, Body лҘј л°ҳнҷҳн•  мҲҳ м—Ҷм—ҲлӢӨ.
              status: 200,
              statusText: "OK",
              headers: {
                "Content-Length": cachedResponse.length,
                "Content-Type": "application/json",
                // к·ёлһҳм„ң м»ӨмҠӨн…Җ н—ӨлҚ”лҘј м¶”к°Җн–ҲлӢӨ.
                "X-SW-Cache-Hit": 1,
                "X-SW-Cache-Type": "POST",
              },
            });
          }

          // мәҗмӢңлҗң лҚ°мқҙн„°к°Җ м—Ҷмқ„ кІҪмҡ° мғҲлЎң мҡ”мІӯн•ңлӢӨ.
          return fetch(fetchEvent.request).then((response) => {
            console.log("Fetching response => ", response.clone());

            // м •мғҒм Ғмқј кІҪмҡ°л§Ң IndexedDB м—җ м ҖмһҘн•ңлӢӨ.
            if (200 <= response.status && response.status < 400) {
              // мқҙ мһ‘м—…мқҖ л№„лҸҷкё°м§Җл§Ң көімқҙ кё°лӢӨлҰ¬м§Җ м•Ҡм•„лҸ„ лҗңлӢӨ.
              response
                .clone()
                .text()
                .then((textResponse) => {
                  console.log("Caching response => ", textResponse);
                  return localforage.setItem(sha1Hash, textResponse);
                });
            }

            return response;
          });
        });
      })
      .catch(() => fetch(fetchEvent.request)),
  );
};

// лҰ¬мҠӨл„ҲлҘј л“ұлЎқн•ҙмӨҖлӢӨ.
self.addEventListener("fetch", postRequestFetchListener);
