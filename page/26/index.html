function insertionSort(unsortedList) {
  const len = unsortedList.length;
  for (let i = 1; i < len; i++) {
    // Copy of the current element.
    const tmp = unsortedList[i];
    // Check through the sorted part and compare with the number in tmp. If large, shift the number
    for (var j = i - 1; j >= 0 && unsortedList[j] > tmp; j--) {
      // Shift the number
      unsortedList[j + 1] = unsortedList[j];
    }
    // Insert the copied number at the correct position
    // in sorted part.
    unsortedList[j + 1] = tmp;
  }
}

function selectionSort(items) {
  const length = items.length;
  for (let i = 0; i < length - 1; i++) {
    // Number of passes
    // min holds the current minimum number position for each pass; i holds the Initial min number
    let min = i;
    // Note that j = i + 1 as we only need to go through unsorted array
    for (let j = i + 1; j < length; j++) {
      // Compare the numbers
      if (items[j] < items[min]) {
        // Change the current min number position if a smaller num is found
        min = j;
      }
    }
    if (min != i) {
      // After each pass, if the current min num != initial min num, exchange the position.
      // Swap the numbers
      const tmp = items[i];
      items[i] = items[min];
      items[min] = tmp;
    }
  }
}

  if (list.length < 2) {
    return list;

  const smaller = list.filter((item) => item < pivot);
  const bigger = list.filter((item) => item > pivot);

  return [...quickSort(smaller), pivot, ...quickSort(bigger)];
};

function LinkedList() {
  this.head = null;
}

LinkedList.prototype.push = function (val) {
  const node = {
    value: val,
    next: null,
  };

  if (!this.head) {
    this.head = node;
  } else {
    current = this.head;
    while (current.next) {
      current = current.next;
    }
    current.next = node;
  }
};

// Implement a Graph
// basic operations:
//  - add vertex (node)
//  - add edge (node -> node)

function GraphNode(val) {
  this.val = val;
  this.edges = {};
}

function Graph() {
  this.vertices = {};
}

// O(1) operation
Graph.prototype.addVertex = function (val) {
  // add vertex only if it does not exist.
  if (!this.vertices[val]) {
    this.vertices[val] = new GraphNode(val);
  }
};

// O(E) operation - edges
Graph.prototype.removeVertex = function (val) {
  if (this.vertices[val]) {
    // once you remove a vertex, you need to remove all edges pointing
    // to the vertex.
    delete this.vertices[val];

    Object.keys(this.vertices).forEach(
      function (key, index) {
        if (this.vertices[key].edges[val]) {
          delete this.vertices[key].edges[val];
        }
      }.bind(this),
    );
  }
};

// O(1) operation
Graph.prototype.getVertex = function (val) {
  return this.vertices[val];
};

// O(1) operation
Graph.prototype.addEdge = function (start, end) {
  // check to see if vertices exists.
  // if it exists, set the edges and be done.
  if (this.vertices[start] && this.vertices[end]) {
    // check to see if edge exists, if it does, increment it's weight
    if (this.vertices[start].edges[end]) {
      this.vertices[start].edges[end].weight += 1;
    } else {
      // edge does not exist, set weight to 1.
      this.vertices[start].edges[end] = { weight: 1 };
    }
  }
};

// O(1) operation
Graph.prototype.removeEdge = function (start, end) {
  if (this.vertices[start] && this.vertices[end]) {
    if (this.vertices[start].edges[end]) {
      delete this.vertices[start].edges[end];
    }
  }
};

// O(1) operation
Graph.prototype.getEdge = function (start, end) {
  return this.vertices[start].edges[end] || null;
};

Graph.prototype.neighbors = function (val) {
  return this.vertices[val] ? this.vertices[val].edges : null;
};

graph.addVertex(5);
graph.addVertex(2);
graph.addVertex(6);
graph.addVertex(7);
graph.addEdge(2, 5);
graph.addEdge(6, 7);
graph.addEdge(7, 5);

console.log(graph.getEdge(6, 7));
graph.removeVertex(5);
console.log(graph.getEdge(2, 5));
console.log(graph.neighbors(6));

function removeEmoji($text) {
    $clean_text = "";
    // Match Emoticons
    $regexEmoticons = '/[\x{1F600}-\x{1F64F}]/u';
    $clean_text = preg_replace($regexEmoticons, '', $text);
    // Match Miscellaneous Symbols and Pictographs
    $regexSymbols = '/[\x{1F300}-\x{1F5FF}]/u';
    $clean_text = preg_replace($regexSymbols, '', $clean_text);
    // Match Transport And Map Symbols
    $regexTransport = '/[\x{1F680}-\x{1F6FF}]/u';
    $clean_text = preg_replace($regexTransport, '', $clean_text);
    // Match Miscellaneous Symbols
    $regexMisc = '/[\x{2600}-\x{26FF}]/u';
    $clean_text = preg_replace($regexMisc, '', $clean_text);
    // Match Dingbats
    $regexDingbats = '/[\x{2700}-\x{27BF}]/u';
    $clean_text = preg_replace($regexDingbats, '', $clean_text);
    // Match Flags
    $regexDingbats = '/[\x{1F1E6}-\x{1F1FF}]/u';
    $clean_text = preg_replace($regexDingbats, '', $clean_text);
    // Others
    $regexDingbats = '/[\x{1F910}-\x{1F95E}]/u';
    $clean_text = preg_replace($regexDingbats, '', $clean_text);
    $regexDingbats = '/[\x{1F980}-\x{1F991}]/u';
    $clean_text = preg_replace($regexDingbats, '', $clean_text);
    $regexDingbats = '/[\x{1F9C0}]/u';
    $clean_text = preg_replace($regexDingbats, '', $clean_text);
    $regexDingbats = '/[\x{1F9F9}]/u';
    $clean_text = preg_replace($regexDingbats, '', $clean_text);
    return $clean_text;
}

Dynamic cachingвҖӢ

sw.js
const DYNAMIC_CACHE_NAME = "лӢӨмқҙлӮҳлҜ№-мәҗмӢң-мҠӨнҶ лҰ¬м§Җ1";

// fetch eventлҠ” м–ҙл”ҳк°Җм—җм„ң лҰ¬мҶҢмҠӨлҘј к°Җм ёмҳ¬ л•Ң лӘЁл‘җ мӢӨн–үлҗңлӢӨ.
// jsлҘј к°Җм ёмҳӨкұ°лӮҳ мқҙлҜём§ҖлҘј к°Җм ёмҳӨкұ°лӮҳ нҺҳмқҙм§ҖлҘј к°Җм ёмҳӨкұ°лӮҳ л“ұл“ұ
self.addEventListener("fetch", (event) => {
  event.respondWith(
    caches.match(event.request).then((response) => {
      // мәҗмӢңм—җ мһҲмңјл©ҙ repsonseлҘј к·ёлҢҖлЎң лҸҢл ӨмӨҖлӢӨ.
      if (response) {
        return response;
      }

      // м—¬кё°м„ң requestлҘј ліөмӮ¬н•ҙмӨҖлӢӨ.
      // requestлҠ” мҠӨнҠёлҰјмңјлЎң fetch лӢ№ н•ң лІҲл§Ң мӮ¬мҡ©н•ҙм•јн•ҳкё° л•Ңл¬ёмқҙлӢӨ.
      // к·јлҚ° event.requestлЎң л°ӣм•„лҸ„ мӢӨн–үмқҖ лҗңлӢӨ
      const fetchRequest = event.request.clone();

      // if (response) return response кө¬л¬ёмқ„ н•ҳлӮҳлЎң н•©м№  мҲҳлҸ„ мһҲлӢӨ.
      // return response || fetch(fetchRequest)
      return fetch(fetchRequest).then((response) => {
        // мқ‘лӢөмқҙ м ңлҢҖлЎң мҷ”лҠ”м§Җ мІҙнҒ¬н•ңлӢӨ.
        // кө¬кёҖ л¬ём„ңм—җлҠ” лӢӨмқҢкіј к°ҷмқҙ мІҳлҰ¬н•ҳлқјкі  лҗҳм–ҙмһҲлҠ”лҚ°
        // мқҙ кІҪмҡ° Cross Site Requestм—җ лҢҖн•ҙ мәҗмӢұ мІҳлҰ¬лҘј н•  мҲҳк°Җ м—ҶлӢӨ.
        // if(!response || response.status !== 200 || response.type !== 'basic') {
        if (!response) {
          return response;
        }

        // мқ‘лӢөмқҖ кјӯ ліөмӮ¬ н•ҙмӨҳм•јн•ңлӢӨ.
        const responseToCache = response.clone();

        // мәҗмӢң мҠӨнҶ лҰ¬м§ҖлҘј м—ҙкі  м •л§җ мәҗмӢұмқ„ н•ҙмӨҖлӢӨ.
        caches.open(DYNAMIC_CACHE_NAME).then((cache) => {
          cache.put(event.request, responseToCache);
        });

        // м—¬кё°м„ң responseлҘј лӮҙліҙлӮҙмӨҳм•ј мәҗмӢұ мІҳлҰ¬ нӣ„м—җ лҰ¬мҶҢмҠӨлҘј л°ҳнҷҳн•ңлӢӨ.
        return response;
      });
    }),
  );
});

// м„ңл№„мҠӨ мӣҢм»Өк°Җ нҷңм„ұнҷ” лҗ  л•Ң

  // мҳҒкө¬м ҒмңјлЎң к°Җм ёк°Ҳ мәҗмӢң мҠӨнҠёлҰ¬м§Җ нҷ”мқҙнҠёлҰ¬мҠӨнҠё
  const cacheWhiteList = [PRE_CACHE_NAME, DYNAMIC_CACHE_NAME];

  event.waitUntil(
    // мәҗмӢң мҠӨнҶ лҰ¬м§Җмқҳ лӘЁл“  мҠӨнҶ лҰ¬м§ҖлӘ…мқ„ к°Җм ёмҳЁлӢӨ.

      // мәҗмӢңлҘј мӮӯм ңн•ҳлҠ” кұҙ PromiseлҘј л°ҳнҷҳн•ҳлҜҖлЎң Promise.allмқ„ мӮ¬мҡ©н•ҙ лҒқлӮ  мӢңм җмқ„ мһЎм•„м•јн•ңлӢӨ.
      return Promise.all(
        // мқҙ кІ°кіјлҠ” [Promise, Promise...] нҳ•нғңк°Җ лҗҳмӢңкІ лӢӨ.
        cacheNames.map((cacheName) => {
          // к°Ғк°Ғмқҳ мәҗмӢң мҠӨнҶ лҰ¬м§ҖлӘ…мқҙ нҷ”мқҙнҠё лҰ¬мҠӨнҠёмҷҖ к°ҷм§Җ м•Ҡмқ„ кІҪмҡ°
          if (cacheWhitelist.indexOf(cacheName) === -1) {
            // мәҗмӢңлҘј мӮӯм ңн•ҳлҠ” PromiseлҘј л°°м—ҙм—җ м¶”к°Җн•ңлӢӨ.
            return caches.delete(cacheName);
          }
        }),
      );
    }),
  );

  // activate мӢңм—җлҠ” clients claim л©”мҶҢл“ңлҘј нҳём¶ңн•ҙм„ң
  // лёҢлқјмҡ°м Җм—җ лҢҖн•ң м ңм–ҙк¶Ңмқ„ к°Җм ёмҷҖм•јн•ңлӢӨ.
  return self.clients.claim();
});

м„ңл№„мҠӨмӣҢм»Өм—җм„ң fetchлЎң мҷёл¶Җ лҰ¬мҶҢмҠӨлҘј к°Җм§Җкі мҳӨл©ҙ opaque responseк°Җ л°ҳнҷҳлҗңлӢӨ. cors м •мұ…мқҙ м„Өм •лҗҳм–ҙ мһҲм§Җ м•Ҡм•„ м•„л¬ҙ м •ліҙлҸ„ к°Җм§Җкі  мҳ¬ мҲҳ м—ҶлҠ” кұҙлҚ°, мқҙлҹ° лҰ¬мҶҢмҠӨл§Ң кіЁлқјм„ң мәҗмӢұмІҳлҰ¬лҘј н•ҳкі  мӢ¶лӢӨл©ҙ request urlмқҙлӮҳ response content-typeмқ„ к°Җм§Җкі  мІҳлҰ¬н•  мҲҳ мһҲлӢӨ.

sw.js
const dynamicCacheStrategy = (event) => {
  // мәҗмӢұ мІҳлҰ¬н•ҳкі  мӢ¶мқҖ content-type
  const cacheContentsTypes = [
    "image/png",
    "image/gif",
    "image/jpeg",
    "application/font-woff",
  ];

  event.respondWith(
    caches.match(event.request).then((response) => {
      const fetchRequest = event.request.clone();

      return (
        response ||
        fetch(fetchRequest)
          .then((response) => {
            if (!response) {
              return response;
            }

            // response headerм—җм„ң content-typeмқ„ к°Җм ёмҷҖ л№„көҗн•ңлӢӨ.
            // м•„лӢҲл©ҙ request.urlмқҙ мәҗмӢұмІҳлҰ¬лҘј н•  мҷёл¶Җ urlмқём§Җ нҷ•мқён•ңлӢӨ.
            if (
              cacheContentsTypes.indexOf(
                response.headers.get("content-type"),
              ) !== -1 ||
              event.request.url.indexOf("external.url") !== -1
            ) {
              caches.open(DYNAMIC_CACHE_NAME).then((cache) => {
                cache.put(event.request, response.clone());
              });
            }

            return response;
          })
          .catch((error) => console.log(error))
      );
    }),
  );
};

мҳӨлҠҳ лӮ м§ңлҘј кө¬н•ҳкё° мң„н•ҙ momentлҘј лЁјм Җ м„Өм№ҳн•ҳмһҗ. winston нҢЁнӮӨм§ҖлҠ” @adonisjs/framework нҢЁнӮӨм§Җм—җ мў…мҶҚлҗңлӢӨ.

config/app.js
const moment = use("moment");
const { config } = use("winston");

module.exports = {
  logger: {
    transport: "file",
    file: {
      driver: "file",
      name: "adonis-app",
      // нҢҢмқјлӘ…мқ„ л°”кҝ”мӨҖлӢӨ
      filename: `adonis_${moment().format("YYYYMMDD")}.log`,
      level: "debug",
      // json лЎңк·ё нҸ¬л§·мқ„ н•ҙм ңн•ҳкі 
      json: false,
      // мӣҗн•ҳлҠ” нҳ•нғңлЎң л°”кҝ”мӨҖлӢӨ
      formatter: ({ level, message, meta }) => {
        const now = moment().format("YYYY-MM-DD HH:mm:ss.SSS");
        // лЎңк·ё л ҲлІЁм—җ мғүмғҒмқ„ м¶”к°Җн•ҳлҠ” мһ‘м—…мқёлҚ°, көімқҙ н•„мҡ”н•ҳм§„ м•ҠлӢӨ
        const logLevel = config.colorize(level, level.toUpperCase());
        const formattedMeta =
          meta && Object.keys(meta).length ? "\n\t" + JSON.stringify(meta) : "";

        return `[${now}] ${logLevel} ${message || ""} ${formattedMeta}`;
      },
    },
  },
};
