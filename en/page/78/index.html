/**
 * [num2han мҲ«мһҗлҘј н•ңкёҖлЎң ліҖнҷҳ]
 * @param  {[integer]} num [мҲ«мһҗ]
 * @return {[string]}      [н•ңкёҖ мҲ«мһҗ]
 * @author http://www.phpschool.com/gnuboard4/bbs/board.php?bo_table=tipntech&wr_id=14981
 */
const num2han = function (num) {
  let i,
    j = 0,
    k = 0;
  const han1 = ["", "мқј", "мқҙ", "мӮј", "мӮ¬", "мҳӨ", "мңЎ", "м№ ", "нҢ”", "кө¬"];
  const han2 = ["", "л§Ң", "м–ө", "мЎ°", "кІҪ", "н•ҙ", "мӢң", "м–‘", "кө¬", "к°„"];
  const han3 = ["", "мӢӯ", "л°ұ", "мІң"];
  let result = "";
  let hangul = String(num);
  let pm = ""; // л¶Җнҳё
  let str = [],
    str2 = "";
  const strTmp = [];

  if (Number(num) === 0) {
    return "мҳҒ";
  }

  if (hangul.substring(0, 1) === "-") {
    pm = "л§Ҳмқҙл„ҲмҠӨ ";
    hangul = hangul.substring(1, hangul.length);
  }

  if (hangul.length > han2.length * 4) {
    return "too much number";
  }

  for (i = hangul.length; i > 0; i = i - 4) {
    str[j] = hangul.substring(i - 4, i);

    for (k = str[j].length; k > 0; k--) {
      strTmp[k] = str[j].substring(k - 1, k) ? str[j].substring(k - 1, k) : "";
      strTmp[k] = han1[parseInt(strTmp[k])];

      if (strTmp[k]) {
        strTmp[k] += han3[str[j].length - k];
      }

      str2 = strTmp[k] + str2;
    }

    str[j] = str2;

    if (str[j]) {
      result = str[j] + han2[j] + result;
    }

    // 4мһҗлҰ¬л§ҲлӢӨ н•ңм№ём”© лқ„кі  ліҙм—¬мӨҖлӢӨ.
    //result = (str[j])? " "+str[j]+han2[j]+result : " " + result;
    j++;
    str2 = "";
  }

  return pm + result;
};

/**

 */
const searchAddr = function () {
  new daum.Postcode({
    oncomplete: function (data) {
      let fullAddr = ""; // мөңмў… мЈјмҶҢ ліҖмҲҳ
      let extraAddr = ""; // мЎ°н•©нҳ• мЈјмҶҢ ліҖмҲҳ
      let engAddr = "";
      let zipcode = "";

      // мӮ¬мҡ©мһҗк°Җ м„ нғқн•ң мЈјмҶҢ нғҖмһ…м—җ л”°лқј н•ҙлӢ№ мЈјмҶҢ к°’мқ„ к°Җм ёмҳЁлӢӨ.
      if (data.userSelectedType === "R") {
        // мӮ¬мҡ©мһҗк°Җ лҸ„лЎңлӘ… мЈјмҶҢлҘј м„ нғқн–Ҳмқ„ кІҪмҡ°
        fullAddr = data.roadAddress;
        zipcode = data.zonecode;
        engAddr = data.roadAddressEnglish;

        //лІ•м •лҸҷлӘ…мқҙ мһҲмқ„ кІҪмҡ° м¶”к°Җн•ңлӢӨ.
        if (data.bname !== "") {
          extraAddr += data.bname;
        }
        // кұҙл¬јлӘ…мқҙ мһҲмқ„ кІҪмҡ° м¶”к°Җн•ңлӢӨ.
        if (data.buildingName !== "") {
          extraAddr +=
            extraAddr !== "" ? ", " + data.buildingName : data.buildingName;
        }
        // мЎ°н•©нҳ•мЈјмҶҢмқҳ мң л¬ҙм—җ л”°лқј м–‘мӘҪм—җ кҙ„нҳёлҘј м¶”к°Җн•ҳм—¬ мөңмў… мЈјмҶҢлҘј л§Ңл“ лӢӨ.
        fullAddr += extraAddr !== "" ? " (" + extraAddr + ")" : "";
      } else {
        // мӮ¬мҡ©мһҗк°Җ м§ҖлІҲ мЈјмҶҢлҘј м„ нғқн–Ҳмқ„ кІҪмҡ°(J)
        fullAddr = data.jibunAddress;

        engAddr = data.jibunAddressEnglish;
      }

      // кө¬лІ„м „мқј кІҪмҡ° getElementById лЎң ліҖкІҪ
      document.querySelector("#zip").value = zipcode;
      document.querySelector("#address").value = fullAddr;
      document.querySelector("#address_eng").value = engAddr;

      document.querySelector("#address_detail").focus();
    },
  }).open();
