Laravel 5.5 - лӢӨнҳ•м„ұ кҙҖкі„

лҢ“кёҖ н…Ңмқҙлё”мқҙ мһҲкі  мқҙ лҢ“кёҖмқҖ м—¬лҹ¬ н…Ңмқҙлё”м—җм„ң мӮ¬мҡ©лҗңлӢӨкі  м№ҳмһҗ. к·ёлҹј лҢ“кёҖ н…Ңмқҙлё”м—җ typeкіј type_idлҘј к°Җм ёк°Җм•јн•  кІғмқҙлӢӨ. мқҙ л•Ң мӮ¬мҡ©н•  мҲҳ мһҲлҠ” кҙҖкі„к°Җ лӢӨнҳ•м„ұ кҙҖкі„(лҰҙл Ҳмқҙм…ҳ)мқёлҚ°, кіөмӢқ л¬ём„ңмқҳ м„ӨлӘ…мқҙ мЎ°кёҲмқҖ л¶ҖмЎұн•ҳлӢӨкі  лҠҗкјҲлӢӨ. нҢҢн—Өміҗліҙмһҗ.

morphToвҖӢ

morphToлҠ” typeкіј type_idлҘј к°Җм§„, м—¬лҹ¬ н…Ңмқҙлё”лЎң м—°кІ°лҗҳм–ҙм•јн•  н…Ңмқҙлё”м—җм„ң мӮ¬мҡ©н•ҳлҠ” лҰҙл Ҳмқҙм…ҳ л©”мҶҢл“ңмқҙлӢӨ. кіөмӢқ л¬ём„ңм—җлҠ” лҚ°мқҙн„°лҘј к°Җм ёмҳЁ л’Ө лҰҙл Ҳмқҙм…ҳмқ„ м—°кІ°н•ҳлҠ” мҳҲмӢңл§Ң мһҲкі , Eager лЎңл”© (With кө¬л¬ёмқ„ мӮ¬мҡ©н•ҳлҠ” л°©лІ•) нӣ„ specificн•ң н•„л“ңлҘј мӮ¬мҡ©н•ҳкІҢ ліҖкІҪн•ҳлҠ” кІҪмҡ°м—җ лҢҖн•ң м •ліҙлҠ” м—ҶлӢӨ.

кё°ліё л¬ёлІ•вҖӢ

YourModel.php
<?php

class YourModel extends Model {
  ...

  // лӢӨнҳ•м„ұ кҙҖкі„лҘј к°Җм§Ҳ н•ЁмҲҳлҘј dataлЎң м •мқҳн–ҲлӢӨ
  public function data() {
    return $this->morphTo();
  }
}

мқҙл ҮкІҢ м •мқҳмӢңм—җ YourModel::with('data')->get() мңјлЎң нҳём¶ңн•ҳл©ҙ л¶Ҳлҹ¬мҷҖм ём•јлҗҳм§Җл§Ң, н•„л“ңлӘ…, лӘЁлҚёлӘ…мқҙ м •нҷ•н•ҳм§Җ м•Ҡмңјл©ҙ мҝјлҰ¬ нҳём¶ңмЎ°м°Ё лҗҳм§Җ м•ҠлҠ”лӢӨ. (мӢ¬м§Җм–ҙ м—җлҹ¬лҸ„ л°ңмғқн•ҳм§Җ м•ҠлҠ”лӢӨ)

н•„л“ңлӘ… м •мқҳвҖӢ

лЁјм Җ morphToмқҳ мҶҢмҠӨмҪ”л“ңлҘј лҘј мӮҙнҺҙліҙмһҗ.

morphTo
<?php
/**
  * Define a polymorphic, inverse one-to-one or many relationship.
  *
  * @param  string  $name
  * @param  string  $type
  * @param  string  $id
  * @return \Illuminate\Database\Eloquent\Relations\MorphTo
  */
public function morphTo($name = null, $type = null, $id = null)

name, type, idлҘј нҢҢлқјлҜён„°лЎң л°ӣлҠ”лӢӨ. к·ёлҹј нҢҢлқјлҜён„°лҘј л„ҳкІЁліҙмһҗ.

YourModel.php
<?php

class YourModel extends Model {
  ...

  public function data() {
    // morphToмқҳ paremeterлЎң null, нғҖмһ…н•„л“ңлӘ…, нғҖмһ…мқёлҚұмҠӨ н•„л“ңлӘ…мқ„ л„ҳкёҙлӢӨ.
    return $this->morphTo(null, 'type', 'type_idx');
  }
}

м—¬кё°м„ң nameм—” лҸ„лҢҖмІҙ лӯҳ л„Јм–ҙм•ј лҗҳлҠ”кұ°м•ј? лқјкі  мқҳл¬ёмқҙ мғқкёё мҲҳк°Җ мһҲлӢӨ. н•ЁмҲҳ лӮҙм—җм„ң name ліҖмҲҳлҠ” $this->getMorphs(Str::snake($name), $type, $id); м—җл§Ң л”ұ н•ң лІҲ мӮ¬мҡ©лҗңлӢӨ.

getMorphs н•ЁмҲҳлҘј л”°лқјк°Җліҙмһҗ.

getMorphs
<?php
protected function getMorphs($name, $type, $id) {
  // $typeкіј $idк°Җ лӘ…мӢңлҗҳл©ҙ к·ё к°’мқ„ лЁјм Җ л°ҳнҷҳн•ңлӢӨ.
  return [$type ?: $name.'_type', $id ?: $name.'_id'];
}

мЈјм„қмІҳлҹј typeкіј idк°Җ лӘ…мӢңлҗҳл©ҙ nameк°’мқҖ мӮ¬мҡ©лҗҳм§Җ м•ҠлҠ” м“°л Ҳкё° к°’мқҙ лҗҳм–ҙлІ„лҰ°лӢӨ. л”°лқјм„ң nullлЎң л„ҳкІЁмЈјл©ҙ лҗңлӢӨ.

нғҖмһ…-лӘЁлҚё л°”мқёл”©вҖӢ

нғҖмһ…кіј мқёлҚұмҠӨлҘј лӘ…мӢңн•ҳл©ҙ л“ңл””м–ҙ мҳӨлҘҳл©”м„ём§Җк°Җ л…ём¶ңлҗңлӢӨ. лӮҙк°Җ м •мқҳн•ң typeлӘ…мқ„ к°Җм§„ Classк°Җ м—ҶлӢӨ лқјлҠ” лӮҙмҡ©мқёлҚ°, мқҙм ң нғҖмһ…кіј лӘЁлҚёмқ„ м—°кІ°мӢңмјңліҙмһҗ.

мқҙ л•Ң мӮ¬мҡ©н•  мҲҳ мһҲлҠ” л©”мҶҢл“ңк°Җ кіөмӢқ л¬ём„ңм—җм„ң мЎ°кёҲ мҠӨнҒ¬лЎӨмқ„ лӮҙлҰ¬л©ҙ мһҲлҠ” Custom Polymorphic Typesм—җ мһҳ м„ӨлӘ…лҗҳм–ҙ мһҲлӢӨ.

н•ҳм§Җл§Ң л“ұлЎқн•ҳлҠ” л¶Җл¶„м—җ лҢҖн•ң м„ӨлӘ…мқҙ **You may register the morphMap in the boot function of your AppServiceProvider or create a separate service provider if you wish.**лқјкі  лҗҳм–ҙмһҲлӢӨ. мҰү AppServiceProviderм—җ л„ЈлҚҳм§Җ Service ProviderлЎң мғқм„ұмқҙлӢӨ.

н•ң лӘЁлҚём—җл§Ң м“ё кұҙлҚ° м „мІҙм—җ л“ұлЎқмқ„ н•  н•„мҡ”к°Җ м—ҶмңјлӢҲ, мӮ¬мҡ©н•  лӘЁлҚём—җ кё°лҠҘмқ„ л„Јм–ҙліҙмһҗ.

YourModel.php
// Relationмқ„ мӮ¬мҡ©н•ҙм•јн•ңлӢӨ.
use Illuminate\Database\Eloquent\Relations\Relation;

<?php
class YourModel extends Model {
  // мқҙ л©”мҶҢл“ңлҠ” лӘЁлҚёмқҙ initializeлҗ  л•Ң мӢӨн–үлҗңлӢӨ.
  protected static function boot() {
    parent::boot();

    // м—¬кё°м—җ нғҖмһ… лі„лЎң лӘЁлҚёмқ„ л°”мқёл”©н•ңлӢӨ.
    Relation::morphMap([
      // typeмқҙ productмқј кІҪмҡ° idлҠ” product_idлҘј к°ҖлҰ¬нӮЁлӢӨ.
      'product' => 'App\Models\Product',
      // typeмқҙ orderмқј кІҪмҡ° idлҠ” order_idлҘј к°ҖлҰ¬нӮЁлӢӨ.
      'order' => 'App\Models\Order'
    ]);
  }

  public function data() {
    // morphToмқҳ paremeterлЎң null, нғҖмһ…н•„л“ңлӘ…, нғҖмһ…мқёлҚұмҠӨ н•„л“ңлӘ…мқ„ л„ҳкёҙлӢӨ.
    return $this->morphTo(null, 'type', 'type_idx');
  }
}

мҷ„лІҪн•ҙмЎҢлӢӨ. мқҙм ң мҳӨлҘҳ м—Ҷмқҙ мӢӨн–үлҗҳлҠ” кІғмқ„ нҷ•мқён•  мҲҳ мһҲлӢӨ.

morphOneвҖӢ

л¬ём„ң мғҒм—җлҠ” м„ӨлӘ… лҗҳм§Җ м•ҠмқҖ morphOne мқҙлһҖ л©”мҶҢл“ңлҸ„ мһҲлӢӨ. morphManyлҠ” кҙҖкі„к°Җ м„Өм •лҗң к°’мқ„ л°°м—ҙлЎң л°ҳнҷҳн•ҳм§Җл§Ң morphOneмқҖ н•ҳлӮҳмқҳ лҚ°мқҙн„°лЎң л°ҳнҷҳн•ңлӢӨ. (hasOneкіј hasManyмІҳлҹј)

лӢӨлҢҖлӢӨ лӢӨнҳ•м„ұ кҙҖкі„ л©”мҶҢл“ң (morphToMany, morphedByMany)мқҳ кІҪмҡ°лҠ” лӮҳмӨ‘м—җ мӮ¬мҡ©н•ҳкІҢ лҗҳл©ҙ м •лҰ¬н•ҙм•јкІ лӢӨ. Relation л©”мҶҢл“ңл“ӨмқҖ л¬ём„ңлҘј лҢҖм¶© нӣ‘кі  API DocsлҘј м§Ғм ‘ к№ҢліҙлҠ”кІҢ, нҡЁмңЁм Ғмқё кІғ к°ҷлӢӨ.
