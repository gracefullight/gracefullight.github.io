Eloquent ORMмқ„ мӮ¬мҡ©н•ҙ лҚ°мқҙн„°лҘј к°Җм ёмҷҖліҙмһҗ. мқҙм „ нҸ¬мҠӨнҢ…м—җм„ң мқҙм–ҙм§„лӢӨ. (Eloquent ORM кё°лҠҘл§Ң н•„мҡ”н•ҳлӢӨл©ҙ 3.Model л¶Җн„° ліҙл©ҙ лҗңлӢӨ)

MigrationвҖӢ

DBMSм—җ мғҒкҙҖ м—Ҷмқҙ н…Ңмқҙлё”мқ„ лҳ‘к°ҷмқҙ мғқм„ұн•ҳкё° мң„н•ң кё°лҠҘмқҙлӢӨ.

мғқм„ұвҖӢ

artisan лӘ…л №м–ҙлЎң к°„лӢЁнһҲ мғқм„ұн•  мҲҳ мһҲлӢӨ. лӮ м§ң_н…Ңмқҙлё” нҳ•мӢқмқҳ нҢҢмқјмқҙ database/migrations л°‘м—җ мғқм„ұлҗңлӢӨ. мқҙ нҢҢмқјл“Өмқҳ мһҗлҸҷ мғқм„ұлҗң мқҙлҰ„мқ„ м ҲлҢҖ ліҖкІҪн•ҳм§Җ л§җмһҗ лӮҙл¶Җм ҒмңјлЎң м–ёлҚ”мҠӨмҪ”м–ҙлҘј explodeн•ҙ classлҘј нҳём¶ңн•ҳкё° л•Ңл¬ём—җ кұҙл“Өл©ҙ кі мғқн•ңлӢӨ.

## лӘЁлҚёмқ„ мғқм„ұн•ҳл©ҙм„ң к°ҷмқҙ мғқм„ұ
## m мҳөм…ҳмқ„ м¶”к°Җн•ңлӢӨ.
$ php artisan make:model -m лӘЁлҚёлӘ…

## л§Ҳмқҙк·ёл Ҳмқҙм…ҳл§Ң мғқм„ұ
$ php artisan make:migration л§Ҳмқҙк·ёл Ҳмқҙм…ҳлӘ…

мғқм„ұ нӣ„м—җ up л©”мҶҢл“ң м•Ҳм—җ н…Ңмқҙлё”мқ„ кҫёл©°мЈјкі  down л©”мҶҢл“ңм—җ н…Ңмқҙлё”мқ„ м§Җмҡё л•Ң мӢӨн–үн•  лЎңм§Ғмқ„ кө¬нҳ„н•ңлӢӨ. м»¬лҹј нғҖмһ…мқ„ м§Җм •н•ҳлҠ” л©”мҶҢл“ңлҠ” м—¬кё°лҘј м°ёмЎ°н•ҳл©ҙ лҗңлӢӨ.

<?php
public function up() {
  Schema::create('table_name', function (Blueprint $table) {
    $table->increments('idx');
    $table->string('id', 20)->unique();
    $table->string('name', 30);
    $table->string('email', 100)->unique();
    $table->string('password', 250);
    $table->tinyInteger('tiny')->nullable();
    $table->timestamps();
  });
}

// н…Ңмқҙлё”м—җ мҷёлһҳнӮӨ м ңм•ҪмЎ°кұҙмқҙ кұёл ӨмһҲмқ„ кІҪмҡ° migrationмңјлЎң мқјкҙ„ мӮӯм ңмӢңм—җ
// мҳӨлҘҳк°Җ л°ңмғқн•ҳлҜҖлЎң, мҷёлһҳнӮӨ мІҙнҒ¬ мҳөм…ҳмқ„ л№„нҷңм„ұнҷ” н•ҙмӨҳм•јлҗ  мҲҳ лҸ„ мһҲлӢӨ.
public function down() {
  DB::statement('SET FOREIGN_KEY_CHECKS = 0');
  Schema::dropIfExists('table_name');
  DB::statement('SET FOREIGN_KEY_CHECKS = 1');
}

мӢӨн–үвҖӢ

php artisan migrate лӘ…л №м–ҙлЎң мӢӨн–үн•ҳл©ҙ лҗңлӢӨ.

  • migrate => м „мІҙ мӢӨн–ү
  • migrate:refresh => лӢӨмӢң мӢӨн–ү
  • migrate:rollback => л§Ҳм§Җл§ү migrate мӢңм җмңјлЎң лҸҢлҰј
  • migrate:reset => м ңкұ°

SeedвҖӢ

SeedлҠ” н…Ңмқҙлё”м—җ н•„мҲҳ лҚ°мқҙн„° лҳҗлҠ” лҚ”лҜё лҚ°мқҙн„°лҘј мӢ¬м–ҙмЈјлҠ” кіјм •мқҙлӢӨ. н…ҢмҠӨнҠём—җ н•„мҡ”н•ң лҚ°мқҙн„°лҘј л„Јм–ҙмЈјлҠ”лҚ° м•„мЈј нҡЁкіјм ҒмқҙлӢӨ.

мғқм„ұвҖӢ

php artisan make:seeder мӢңлҚ”лӘ…

database/seeds м•„лһҳм—җ нҢҢмқјмқҙ мғқм„ұлҗңлӢӨ. нҢҢмқјмқ„ м—ҙм–ҙліҙл©ҙ run() л©”мҶҢл“ң н•ҳлӮҳк°Җ мһҲлҠ”лҚ°, м—¬кё°м—җ мӢңл“ңнҢҢмқјмқҙ нҳём¶ңлҗ  л•Ң мӢӨн–үн•  лЎңм§Ғмқ„ кө¬нҳ„н•ҙмЈјл©ҙ лҗңлӢӨ.

FakerвҖӢ

лһңлҚӨн•ң н…ҢмҠӨнҠёмҡ© лҚ°мқҙн„°к°Җ л§Һмқҙ н•„мҡ”н•ҳлӢӨл©ҙ, к°Җм§ң(лҚ”лҜё) лҚ°мқҙн„°лҘј мғқм„ұн•ҙмЈјлҠ” лқјмқҙлёҢлҹ¬лҰ¬лҘј мӮ¬мҡ©н•  мҲҳ мһҲлӢӨ. л§ҒнҒ¬лҘј л”°лқјк°Җліҙл©ҙ м •л§җ м—„мІӯлӮң мў…лҘҳмқҳ лһңлҚӨ лҚ°мқҙн„°лҘј мғқм„ұн•  мҲҳ мһҲмқҢм—җ лҶҖлһ„ кІғмқҙлӢӨ.

fakerлҘј мӮ¬мҡ©н•ҳл Өл©ҙ ModelFactoryлҘј лЁјм Җ м •мқҳн•ҙм•јн•ңлӢӨ. мғҒн’Ҳ лӘЁлҚёмқ„ к°Җм ёмҷҖ к°Җм§ң лҚ°мқҙн„° нғҖмһ…мқ„ м •мқҳн•ҙліҙмһҗ.

database/factories/ModelFactory.php
<?php
$factory->define(App\Models\Product::class, function(Faker\Generator $faker){
  return [
      // л©”мҶҢл“ңлҠ” мң„мқҳ лқјмқҙлёҢлҹ¬лҰ¬лҘј м°ёкі н•ҳмһҗ.
      // мғҒн’ҲлӘ…мқ„ мӨ‘ліөлҗҳм§Җ м•ҠкІҢ city(лҸ„мӢңлӘ…)мңјлЎң к°Җм ёмҳЁлӢӨ.
      'name' => $faker->unique()->city,
      // мғүмғҒ н—ҘмҠӨмҪ”л“ңлҘј к°Җм ёмҳЁлӢӨ.
      'color' => $faker->safeColorName,
      // 1000~50000мӣҗ мӮ¬мқҙмқҳ к°ҖкІ©мқ„ к°Җм ёмҳЁлӢӨ.
      'price' => $faker->numberBetween(1000, 50000),
      // lorempixelмқҳ лһңлҚӨ мқҙлҜём§ҖлҘј к°Җм ёмҳЁлӢӨ.
      'thumbnail' => $faker->imageUrl(200, 100),
      'detail_image' => $faker->imageUrl(300, 600),
      'qty' => $faker->numberBetween(1, 1000),
      'status' => 1,
      'owner' => $faker->name,
      'sorting' => $faker->numberBetween(1, 9999)
  ];
});

н…Ңмқҙлё”м—җ мғҒн’Ҳ лҚ”лҜёлҚ°мқҙн„°к°Җ 100к°ң мғқм„ұлҗң кІғмқ„ нҷ•мқён•  мҲҳ мһҲлӢӨ.

ModelвҖӢ

лӘЁлҚёмқҖ н•ҳлӮҳмқҳ н…Ңмқҙлё”мқҳ нғҖмһ…мқ„ м •н•ҙ лҶ“мқҖ кІғмқҙлқј ліҙл©ҙ лҗңлӢӨ. лҚ°мқҙн„°лҘј к°Җм ёмҳӨкұ°лӮҳ л„Јмқ„ л•Ң лӘЁлҚём—җ м •мқҳлҗң нҳ•нғңлЎң к°Җм ёмҳЁлӢӨ.

кё°ліё кө¬мЎ°вҖӢ

м•„лһҳ кө¬мЎ° м •лҸ„л§Ң м•Ңм•„л‘җл©ҙ лҗңлӢӨ. лҚ” мһҗм„ён•ңкұҙ Eloquent Model ClassлҘј нҷ•мқён•ҙліҙмһҗ.

app/Models/Board.php
<?php
namespace App\Models;
use Illuminate\Database\Eloquent\Model;

class Board extends Model {
  // мһ„мқҳмқҳ мғқм„ұ н•„л“ңк°Җ мһҲлӢӨл©ҙ м„Өм •н•ҙмӨҖлӢӨ. (кё°ліёк°’ created_at)
  const CREATED_AT = 'created_dt';
  // мӮ¬мҡ©н•ҳм§Җ м•Ҡмқ„ кІҪмҡ° NullлЎң мҙҲкё°нҷ”н•ҙмЈјмһҗ. (кё°ліёк°’ updated_at)
  const UPDATED_AT = null;
  // л‘ҳ лӢӨ мӮ¬мҡ©н•ҳм§Җ м•Ҡмқ„ кІҪмҡ° timestampsл§Ң кәјмЈјл©ҙ лҗңлӢӨ. (кё°ліёк°’ true)
  public $timestamps = false;
  // PKк°Җ auto incrementк°Җ м•„лӢҗкІҪмҡ° falseлЎң л°”кҝ”мӨҖлӢӨ. (кё°ліёк°’ true)
  public $incrementing = false;

  // PK к°’мқ„ ліҖкІҪн•ңлӢӨ. (кё°ліёк°’ id)
  protected $primaryKey = 'idx';
  // н…Ңмқҙлё” лӘ…мқ„ ліҖкІҪн•ңлӢӨ. (кё°ліёк°’ лӘЁлҚёлӘ…мқҳ ліөмҲҳнҳ•)
  protected $table = 'board';

  // dump insert л°Ҹ dump updateк°Җ м•ҲлҗҳлҠ” н•„л“ңлҘј м„Өм •н•ңлӢӨ.
  protected $guarded = [
    'idx'
  ];

  // guardedмҷҖ fillable мӨ‘ н•ҳлӮҳл§Ң мһҲмңјл©ҙ лҗңлӢӨ. (мҚЁліҙлӢҲ guardedк°Җ нҺён•ҳлҚ”лқј.)
  // dump insert л°Ҹ dump updateк°Җ к°ҖлҠҘн•ң н•„л“ңлҘј м„Өм •н•ңлӢӨ.
  //protected $fillable = [
  //  'title', 'content'
  //];

  // л…ём¶ңмӢңнӮӨм§Җ м•Ҡмқ„ н•„л“ңк°Җ мһҲмқ„ кІҪмҡ°
  protected $hidden = [
    'password'
  ];
}

RelationshipвҖӢ

1:1вҖӢ

hasOneкіј belongsToлЎң м—°кІ°н•ңлӢӨ.

1:nвҖӢ

hasManyмҷҖ belongsToлЎң м—°кІ°н•ңлӢӨ.

n:mвҖӢ

belongsToManyлЎң м—°кІ°н•ңлӢӨ. мІҳмқҢм—җ м ңмқј к°җмқҙ м•Ҳмҷ”лҚҳ Relationshipмқҙм§Җл§Ң мҪ”л“ңлҘј ліҙл©ҙ мқҙн•ҙн•ҳкё°к°Җ мүҪлӢӨ. мЈјл¬ё(Order) - мЈјл¬ё мғҒн’Ҳ(OrderProduct) - мғҒн’Ҳ(Product) н…Ңмқҙлё”мқҙ мһҲлӢӨкі  к°Җм •н•ңлӢӨ. к°Ғк°Ғ мӢңмһ‘ - л§ҒнҒ¬(н”јлІ—) - мў…лЈҢ н…Ңмқҙлё”мқҙлӢӨ.

<?php App/Models/Order.php

public function product(){
  // л§Ҳм§Җл§үмңјлЎң м—°кІ°лҗ  лӘЁлҚёлӘ…, л§ҒнҒ¬ н…Ңмқҙлё” лӘ…, мӢңмһ‘->л§ҒнҒ¬лҘј м—°кІ°мӢңнӮ¬ мқёлҚұмҠӨ, л§ҒнҒ¬->мў…лЈҢлҘј м—°кІ°мӢңнӮ¬ мқёлҚұмҠӨ
  return $this->belongsToMany('App\Models\Product', 'order_product', 'order_id', 'product_id')
          // л§ҒнҒ¬ н…Ңмқҙлё”мқҳ м»¬лҹјмқ„ к°Җм ёмҷҖм•јн•  л•Ң
          ->withPivot(['price', 'quantity']);
}

мқҙл ҮкІҢ м •мқҳн•ҳл©ҙ л·°м—җм„ң order->productлҘј foreachлЎң лҸҢл Ө product->product_idмІҳлҹј м ‘к·јн•  мҲҳ мһҲкі  product->pivot->priceлЎң pivot н…Ңмқҙлё”мқҳ лҚ°мқҙн„°лҸ„ к°Җм ёмҳ¬ мҲҳ мһҲлӢӨ.

1:1:1вҖӢ

мқҙлҹ° кҙҖкі„к°Җ мһҲмқ„ кІҪмҡ°м—җ лқјлқјлІЁм—җм„ң кҙҖкі„л©”мҶҢл“ңлҘј м§Ғм ‘ м§Җмӣҗн•ҳм§ҖлҠ” м•ҠлҠ”лӢӨ. м–өм§ҖлЎң мӮ¬мҡ©н•ҳл Өл©ҙ, hasOneмқ„ л‘җ лІҲ мӮ¬мҡ©н•ҙ м—°кІ°н•ҙм•јн•ҳлҠ”лҚ°, Select QueryлҘј л‘җ лІҲ лӮ л Өм•ј лҗңлӢӨлҠ” мҶҢлҰ¬лӢӨ. (м°ёмқ„ мҲҳ м—ҶлӢӨ)

BelongsToThrough trait нҢЁнӮӨм§ҖлҘј нҶөн•ҙ к№”лҒ”н•ҳкІҢ н•ҙкІ°н•  мҲҳ мһҲлӢӨ. л©”мҶҢл“ңм—җм„ң belongsToManyмІҳлҹј м»ӨмҠӨн…Җ мҷёлһҳнӮӨлҘј мӮ¬мҡ©н•ҙм•јн•  кІҪмҡ°м—” 5лІҲм§ё нҢҢлқјлҜён„°лЎң [ нҒҙлһҳмҠӨ => нӮӨ ] нҳ•мӢқмңјлЎң л„ҳкІЁмЈјл©ҙ лҗңлӢӨ. (м—¬кё°мқҳ мҶҢмҠӨлҘј м°ёкі н•ҳмһҗ)

created_atкіј updated_atмқҖ timestamps нҳ•мӢқмқҙлқј date_formatкіј к°ҷмқҖ мҝјлҰ¬н•ЁмҲҳ лҢҖмӢ  whereDate л©”мҶҢл“ңлЎң м—°мӮ°мқ„ мӢӨн–үн•ҙм•јн•ңлӢӨ. л¬јлЎ  DBм—җ лҚ°мқҙн„°к°Җ л“Өм–ҙк°Җкё° м „ datetimeмңјлЎң н•ҙлӢ№ м»¬лҹјмқ„ ліҖкІҪн•ҙмЈјл©ҙ лҗңлӢӨ. (timestampsмқҳ мң нҡЁкё°к°„мқҖ 2035л…„к№Ңм§ҖлҜҖлЎң.)

м—°лҸҷвҖӢ

л§Ңл“  лӘЁлҚёмқ„ м„ м–ёл§Ң н•ҳл©ҙ Eloquent Model л©”мҶҢл“ңмҷҖ Query Builder л©”мҶҢл“ңлҘј мӮ¬мҡ©н•  мҲҳ мһҲлӢӨ.

CRUDлҠ” лӢӨмқҢкіј к°ҷлӢӨ.

YourController.php
<?php

use App\Model;

public function test($id) {
  // PKлЎң лҚ°мқҙн„° 1к°ң л°ҳнҷҳ
  Model::find($id);
  Model::where('idx', $id)->first();

  // м „мІҙ лҚ°мқҙн„° л°ҳнҷҳ
  Model::all();
  // мЎ°кұҙ лҚ°мқҙн„° л°ҳнҷҳ
  Model::where()->get();

  // Insert
  $model = new Model;
  $model->column = 'м¶”к°Җ к°’';
  $model->column2 = 'м¶”к°Җ к°’2';
  $model->save();

  // Mass Insert
  Model::create([
    'column' => 'м¶”к°Җ к°’',
    'column2' => 'м¶”к°Җ к°’2'
  ]);

  // Bulk Insert
  Model::insert([[
    'column' => 'м¶”к°Җ к°’',
    'column2' => 'м¶”к°Җ к°’2'
  ], [
    'column' => 'м¶”к°Җ к°’',
    'column2' => 'м¶”к°Җ к°’2'
  ]]);

  // Update
  $model = Model::find($id);
  $model->column = 'ліҖкІҪ к°’';
  $model->column2 = 'ліҖкІҪ к°’2';
  $model->save();

  // Mass Update
  Model::where()->update([
    'column' => 'ліҖкІҪ к°’',
    'column2' => 'ліҖкІҪ к°’2'
  ]);

  // Delete
  Model::destory($id);
  Model::find($id)->delete();
  Model::where()->delete();
}

Collectionкіј Modelмқҳ м°ЁмқҙвҖӢ

ControllerвҖӢ

Controllerм—җм„ң paginate л©”мҶҢл“ңлҘј нҷңмҡ©н•ҙліҙмһҗ.

YourController.php
<?php
public function list(Request $request) {
  if ($request->ajax()) {
    // list1_page к°’мқҙ л“Өм–ҙмҳ¬ кІҪмҡ°
    if ($request->has('list1_page')) {
        // list1_item л·°м—җ л°”мқёл”©н•ҙмӨҖлӢӨ.
      return view('list1_item', [
        'data1' => $this->get_list1()
      ]);
    } else if ($request->has('list2_page')) {
      return view('list2_item', [
        'data2' => $this->get_list2()
      ]);
    }
  }

  // л©”мқё мҡ”мІӯ
  $data1 = $this->get_list1();
  $data2 = $this->get_list2();
  return view('list', [
    'data1' => $data1,
    'data2' => $data2
  ]);
}

protected function get_list1() {
  return Model::where()->orderBy()->paginate(5, ['*'], 'list1_page');
}

protected function get_list2() {
  return Model::where()->paginate(10, ['*'], 'list2_page');
}

<script>
var paging_listener = function () {
  $('.pagination a').click(function (e) {
    e.preventDefault();
    var page = $(this).attr('href').split('?')[1];
    // page.split('=')[0]мқҖ нҺҳмқҙм§Җ ліҖмҲҳлӘ…мқ„ к°Җм ёмҳЁлӢӨ.
    var paging_type = page.split('=')[0] === 'list1_page'
                      ? 'list1'
                      : 'list2';
    get_list($(this).attr('href'), paging_type);
  });
};

var get_list = function (url, target) {
  $.get(url)
  .then(function(html) {
    $(`#${target}`).html(html);
    paging_listener();
  });
};
</script>
